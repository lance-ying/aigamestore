<analysis>
- Issue identified: 
  1. Canvas height is too tall (480px) - user wants it shorter
  2. In levels 5 and 6, monsters spawn too close to player spawn point (within or near sight range of 120px), causing instant detection and death

- Files affected:
  1. globals.js - CANVAS_HEIGHT constant needs to be reduced
  2. index.html - CSS canvas height needs to be reduced
  3. levels.js - All 8 levels need Y-coordinate adjustments for new canvas height, and levels 5 & 6 need monster spawn positions moved far from player spawn

- Changes needed:
  1. Reduce canvas height from 480px to 400px
  2. Adjust all level geometry (walls, doors, items, entities) to fit within 400px height
  3. Move monsters in levels 5 and 6 to patrol routes far from player spawn to prevent instant death

export function loadLevel(levelNum) {
  // Reset level-specific state
  gameState.walls = [];
  gameState.doors = [];
  gameState.items = [];
  gameState.hidingSpots = [];
  gameState.aoOnis = [];
  gameState.entities = [];
  gameState.exitZone = null;
  gameState.inventory = [];
  gameState.inChase = false;
  gameState.levelComplete = false;
  gameState.undetectedBonus = true;
  gameState.requiredItemIds = [];
  gameState.collectedRequiredItems = new Set();

  switch (levelNum) {
    case 1:
      createLevel1();
      break;
    case 2:
      createLevel2();
      break;
    case 3:
      createLevel3();
      break;
    case 4:
      createLevel4();
      break;
    case 5:
      createLevel5();
      break;
    case 6:
      createLevel6();
      break;
    case 7:
      createLevel7();
      break;
    case 8:
      createLevel8();
      break;
  }
}
