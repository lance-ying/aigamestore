// render.js - Rendering functions
import { gameState, GAME_PHASES, CANVAS_WIDTH, CANVAS_HEIGHT } from './globals.js';
import { LEVEL_CONFIGS, getColorName } from './levels.js';

const CELL_SIZE = 35; // Changed from 45 to 35 to fit within 400px height.
const TOP_UI_SAFE_HEIGHT = 90; // Height reserved for top UI (score, moves, level, objectives)

function getGridOffsets() {
  const gridWidth = gameState.gridCols * CELL_SIZE;
  const gridHeight = gameState.gridRows * CELL_SIZE;
  
  // Center horizontally
  const x = (CANVAS_WIDTH - gridWidth) / 2;
  
  // Center vertically within the space *below* the top UI
  const availableHeightForGrid = CANVAS_HEIGHT - TOP_UI_SAFE_HEIGHT;
  let y = TOP_UI_SAFE_HEIGHT + (availableHeightForGrid - gridHeight) / 2;
  
  // Ensure the grid doesn't go off screen if it's too tall for the available space
  y = Math.max(TOP_UI_SAFE_HEIGHT, y);
  y = Math.min(CANVAS_HEIGHT - gridHeight, y);

  return { x, y };
}

export function calculateDotPosition(gridX, gridY) {
  const offsets = getGridOffsets();
  return {
    x: offsets.x + gridX * CELL_SIZE + CELL_SIZE / 2,
    y: offsets.y + gridY * CELL_SIZE + CELL_SIZE / 2
  };
}

export function renderStartScreen(p) {
  p.background(20, 25, 35);

  p.textSize(48);
  p.fill(100, 255, 100);
  p.text("press enter to begin", CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2 - 80); // Adjusted Y

  p.text(description, CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2 - 20); // Adjusted Y

  p.fill(150, 150, 150);

  p.text(instructions, CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2 + 60); // Adjusted Y

  }
}

export function renderPlayingScreen(p) {
  p.background(20, 25, 35);
  
  // Draw UI
  renderUI(p);
  
  const offsets = getGridOffsets();
  const startX = offsets.x;
  const startY = offsets.y;
  const gridWidth = gameState.gridCols * CELL_SIZE;
  const gridHeight = gameState.gridRows * CELL_SIZE;
  
  // Draw grid lines
  p.stroke(60, 60, 80);
  p.strokeWeight(1);
  for (let row = 0; row <= gameState.gridRows; row++) {
    const y = startY + row * CELL_SIZE;
    p.line(startX, y, startX + gridWidth, y);
  }
  for (let col = 0; col <= gameState.gridCols; col++) {
    const x = startX + col * CELL_SIZE;
    p.line(x, startY, x, startY + gridHeight);
  }
  
  // Update and draw dots
  for (let row = 0; row < gameState.gridRows; row++) {
    for (let col = 0; col < gameState.gridCols; col++) {
      const dot = gameState.grid[row][col];
      if (dot) {
        const pos = calculateDotPosition(col, row);
        dot.setTarget(pos.x, pos.y, dot.isClearing ? 0 : CELL_SIZE * 0.78); // Use CELL_SIZE for dot size
        dot.update();
        
        if (dot.alpha > 0) {
          renderDot(p, dot);
        }
      }
    }
  }
  
  // Draw cursor
  renderCursor(p);
  
  // Draw path
  renderPath(p);
  
  // Update and draw particles
  updateAndRenderParticles(p);
  
  // Draw paused indicator
  if (gameState.gamePhase === GAME_PHASES.PAUSED) {
    p.noStroke();
    p.fill(255, 255, 255);
    p.textAlign(p.RIGHT, p.TOP);
    p.textSize(16);
    p.text("PAUSED", CANVAS_WIDTH - 10, 10);
  }
}

function renderUI(p) {
  p.noStroke();
  p.fill(255, 255, 255);
  p.textAlign(p.LEFT, p.TOP);
  p.textSize(16);
  p.text("SCORE: " + gameState.score, 10, 10);
  
  p.textAlign(p.RIGHT, p.TOP);
  p.text("MOVES: " + gameState.currentMoves, CANVAS_WIDTH - 10, 10);
  
  p.textAlign(p.CENTER, p.TOP);
  p.text("LEVEL: " + gameState.currentLevel, CANVAS_WIDTH / 2, 10);
  
  // Objectives
  p.textSize(12);
  p.textAlign(p.LEFT, p.TOP);
  let yOffset = 40;
  for (let key in gameState.levelObjectives) {
    const obj = gameState.levelObjectives[key];
    const progress = obj.current + "/" + obj.target;
    const text = obj.description + ": " + progress;
    const color = obj.current >= obj.target ? [100, 255, 100] : [200, 200, 200];
    p.fill(...color);
    p.text(text, 10, yOffset);
    yOffset += 16;
  }
}

function renderDot(p, dot) {
  if (dot.size <= 0) return;
  
  p.noStroke();
  
  // Draw main body
  p.fill(...dot.color, dot.alpha);
  p.circle(dot.x, dot.y, dot.size);
  
  // Draw elemental icon
  p.fill(255, 255, 255, dot.alpha * 0.9);
  const colorName = getColorName(dot.color);
  
  if (dot.type === 'anchor') {
    drawIcon(p, dot.x, dot.y, dot.size, 'anchor');
  } else {
    drawIcon(p, dot.x, dot.y, dot.size, colorName);
  }
  
  // Highlight if selected
  if (dot === gameState.currentSelectedDot) {
    p.stroke(255, 255, 100);
    p.strokeWeight(4);
    p.noFill();
    p.circle(dot.x, dot.y, dot.size + 8);
  }
  
  // Highlight if in path
  if (gameState.currentPath.includes(dot) && dot !== gameState.currentSelectedDot) {
    p.stroke(255, 255, 255, 150);
    p.strokeWeight(2);
    p.noFill();
    p.circle(dot.x, dot.y, dot.size + 4);
  }
}

function drawIcon(p, x, y, size, type) {
  const s = size * 0.4;
  p.push();
  p.translate(x, y);
  p.noStroke();
  
  switch (type) {
    case 'red': // Fire
      p.beginShape();
      p.vertex(0, -s);
      p.bezierVertex(s*0.5, -s*0.5, s*0.8, s*0.5, 0, s*0.8);
      p.bezierVertex(-s*0.8, s*0.5, -s*0.5, -s*0.5, 0, -s);
      p.endShape();
      break;
      
    case 'blue': // Water
      p.beginShape();
      p.vertex(0, -s*0.8);
      p.bezierVertex(s, 0, 0, s, 0, s);
      p.bezierVertex(0, s, -s, 0, 0, -s*0.8);
      p.endShape();
      break;
      
    case 'green': // Earth/Leaf
      p.beginShape();
      p.vertex(0, -s);
      p.bezierVertex(s*0.8, -s*0.2, s*0.2, s*0.8, 0, s);
      p.bezierVertex(-s*0.2, s*0.8, -s*0.8, -s*0.2, 0, -s);
      p.endShape();
      break;
      
    case 'yellow': // Lightning
      p.beginShape();
      p.vertex(s*0.2, -s);
      p.vertex(-s*0.4, 0);
      p.vertex(0, 0);
      p.vertex(-s*0.2, s);
      p.vertex(s*0.4, 0);
      p.vertex(0, 0);
      p.endShape(p.CLOSE);
      break;
      
    case 'purple': // Magic/Star
      p.beginShape();
      for (let i = 0; i < 5; i++) {
        const angle = -Math.PI/2 + (Math.PI * 2 * i) / 5;
        const angle2 = angle + Math.PI / 5;
        p.vertex(Math.cos(angle) * s, Math.sin(angle) * s);
        p.vertex(Math.cos(angle2) * s * 0.4, Math.sin(angle2) * s * 0.4);
      }
      p.endShape(p.CLOSE);
      break;
      
    case 'orange': // Explosion/Sun
      p.circle(0, 0, s * 0.8);
      p.stroke(255, 255, 255, 200);
      p.strokeWeight(2);
      p.noFill();
      p.line(-s, -s, s, s);
      p.line(s, -s, -s, s);
      break;
      
    case 'anchor':
      p.fill(255);
      p.rect(-s/4, -s/2, s/2, s*0.8);
      p.circle(0, s/2, s);
      p.fill(80, 80, 80); // Hole in anchor
      p.circle(0, s/2, s/3);
      break;
      
    default:
      p.circle(0, 0, s);
  }
  p.pop();
}

function renderCursor(p) {
  const pos = calculateDotPosition(gameState.cursorX, gameState.cursorY);
  p.stroke(180, 180, 180);
  p.strokeWeight(2);
  p.noFill();
  p.rect(pos.x - CELL_SIZE / 2, pos.y - CELL_SIZE / 2, CELL_SIZE, CELL_SIZE);
}

function renderPath(p) {
  if (gameState.currentPath.length > 1) {
    p.stroke(255, 255, 255, 200);
    p.strokeWeight(5);
    p.noFill();
    p.beginShape();
    for (let dot of gameState.currentPath) {
      const pos = calculateDotPosition(dot.gridX, dot.gridY);
      p.vertex(pos.x, pos.y);
    }
    p.endShape();
  }
}

// --- Particle System for Elemental Powers ---

export function spawnParticles(x, y, color) {
  const colorName = getColorName(color);
  const particleCount = 12;
  
  for (let i = 0; i < particleCount; i++) {
    gameState.particles.push(new Particle(x, y, color, colorName));
  }
}

class Particle {
  constructor(x, y, color, type) {
    this.x = x;
    this.y = y;
    this.color = color;
    this.type = type;
    this.life = 1.0;
    this.decay = Math.random() * 0.03 + 0.02;
    
    // Base velocity
    const angle = Math.random() * Math.PI * 2;
    const speed = Math.random() * 2 + 1;
    this.vx = Math.cos(angle) * speed;
    this.vy = Math.sin(angle) * speed;
    
    // Elemental physics overrides
    if (type === 'red') { // Fire: Rise up
      this.vy = -Math.random() * 3 - 1;
      this.vx = (Math.random() - 0.5) * 2;
    } else if (type === 'blue') { // Water: Fall down
      this.vy = Math.random() * 2;
      this.vx = (Math.random() - 0.5) * 3;
    } else if (type === 'yellow') { // Lightning: Fast zaps
      this.vx *= 2.5;
      this.vy *= 2.5;
    } else if (type === 'green') { // Earth: Heavy/Float
      this.decay *= 0.8; // Last longer
    }
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;
    this.life -= this.decay;
    
    // Elemental behavior updates
    if (this.type === 'blue') this.vy += 0.2; // Gravity
    if (this.type === 'green') { // Float/Sway
      this.vx += Math.sin(this.life * 10) * 0.1;
      this.vy += 0.05;
    }
    if (this.type === 'purple') { // Spiral
      const speed = 0.5;
      const dx = -this.vy * speed;
      const dy = this.vx * speed;
      this.vx += dx * 0.1;
      this.vy += dy * 0.1;
    }
    
    return this.life > 0;
  }

  draw(p) {
    p.noStroke();
    const alpha = this.life * 255;
    
    // Blend mode for effect
    if (this.type === 'red' || this.type === 'yellow') {
      p.blendMode(p.ADD);
    }
    
    p.fill(this.color[0], this.color[1], this.color[2], alpha);
    
    const size = this.life * 10;
    
    if (this.type === 'yellow') { // Sparks
      p.push();
      p.translate(this.x, this.y);
      p.rotate(Math.random() * Math.PI * 2);
      p.rect(-size, -2, size * 2, 4);
      p.pop();
    } else if (this.type === 'red') { // Flame bits
      p.circle(this.x, this.y, size);
    } else if (this.type === 'purple') { // Magic dust
      p.circle(this.x, this.y, size * 0.8);
      p.fill(255, 255, 255, alpha);
      p.circle(this.x, this.y, size * 0.3);
    } else {
      p.circle(this.x, this.y, size);
    }
    
    p.blendMode(p.BLEND);
  }
}

function updateAndRenderParticles(p) {
  for (let i = gameState.particles.length - 1; i >= 0; i--) {
    const particle = gameState.particles[i];
    if (particle.update()) {
      particle.draw(p);
    } else {
      gameState.particles.splice(i, 1);
    }
  }
}
