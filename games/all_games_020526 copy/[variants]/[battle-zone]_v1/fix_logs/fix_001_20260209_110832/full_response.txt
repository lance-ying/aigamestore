<analysis>
- Issue identified: The game's pause state currently displays a "PAUSED" text indicator and applies a semi-transparent gray overlay (rectangle) over the screen. The user wants the game to "freeze" (stop updating) without these visual indicators.
- Files affected: `game.js` and `ui.js`.
- Changes needed:
  1. In `game.js`, remove the code that draws the gray overlay in the `PAUSED` case of the `p.draw` loop. Additionally, add the particle drawing loop to the `PAUSED` case to ensure that particles (like muzzle flashes or impact effects) are also "frozen" on screen rather than disappearing when the game is paused.
  2. In `ui.js`, remove the logic in `drawUI` that renders the "PAUSED" text when the game phase is `PAUSED`.

function checkPointInRect(pointX, pointY, rectX, rectY, rectWidth, rectHeight) {
  return pointX >= rectX && pointY >= rectY &&

export function restartGameAndPlay(p, isAutoRestart = false) {
    // Clear any pending auto-restart, whether this is manual or auto
    if (gameState.autoRestartTimeoutId) {
        clearTimeout(gameState.autoRestartTimeoutId);
        gameState.autoRestartTimeoutId = null;
    }
    gameState.autoRestartScheduled = false;

    resetToLevel1(); // Resets current level to 1, calls resetGame() which sets player to null
    gameState.gamePhase = "PLAYING"; // Set phase to PLAYING. The p.draw loop will then create the player and level.

    p.logs.game_info.push({
        "game_status": gameState.gamePhase,
        "data": { "restartType": isAutoRestart ? "auto" : "manual", "level": gameState.currentLevel },
        "framecount": p.frameCount,
        "timestamp": Date.now()
    });
}
