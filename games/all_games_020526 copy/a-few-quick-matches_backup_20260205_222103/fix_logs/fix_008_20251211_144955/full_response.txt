        - Update `handleStunState` to reset `comboHitCount` when recovering to `IDLE`.

    - Update `Player` and `Enemy` classes:

// --- Fighter (Base for Player and AI) ---
export class Fighter extends Entity {
    constructor(x, y, isPlayer = false) {
        super(x, y, 30, 60); // Standard human size
        this.isPlayer = isPlayer;
        
        // Stats
        this.maxHealth = 100;
        this.health = 100;
        this.moveSpeed = 5;
        this.jumpForce = -12;
        this.dashSpeed = 15;
        
        // State
        this.state = "IDLE"; // IDLE, RUN, JUMP, FALL, ATTACK, DASH, STUN, KNOCKDOWN, DEAD
        this.stateTimer = 0;
        this.canDash = true;
        this.hasDoubleJump = true;
        this.attackCooldown = 0;
        this.dashCooldown = 0;
        
        // Combo system
        this.hasHit = false; // Can we cancel?
        this.comboHitCount = 0; // Track consecutive hits for knockdown
        
        // Visuals
        this.animFrame = 0;
        this.currentAttackDirection = null; // Added for visual feedback
    }

    update(p) {
        if (this.health <= 0 && this.state !== "DEAD") {
            this.setState("DEAD");
        }
        
        this.stateTimer++;
        if (this.attackCooldown > 0) this.attackCooldown--;
        if (this.dashCooldown > 0) this.dashCooldown--;

        // Input handling is done in subclasses (Player) or AI logic (Enemy)
        
        // State Machine Update
        switch (this.state) {
            case "IDLE":
            case "RUN":
            case "JUMP":
            case "FALL":
                this.handleMovementState(p);
                break;
            case "ATTACK":
                this.handleAttackState(p);
                break;
            case "DASH":
                this.handleDashState(p);
                break;
            case "STUN":
                this.handleStunState(p);
                break;
            case "KNOCKDOWN":
                this.handleKnockdownState(p);
                break;
            case "DEAD":
                this.vx = 0;
                break;
        }

        updatePhysics(this);
        
        // Collision with Platforms
        gameState.platforms.forEach(platform => {
            resolvePlatformCollision(this, platform);
        });

        // Bounds check (Death pit)
        if (this.y > CANVAS_HEIGHT + 100) {
            this.health = 0;
            this.setState("DEAD");
        }
    }

    handleMovementState(p) {
        // Base physics logic happens in updatePhysics, this controls velocity intent
    }

    handleAttackState(p) {
        // Default empty, override in subclass if needed
    }

    handleDashState(p) {
        // Default empty, override in subclass if needed
    }

    handleStunState(p) {
        // Default recovery logic
        if (this.stateTimer > 20) {
            this.setState("IDLE");
            this.comboHitCount = 0; // Reset combo count on recovery
        }
    }

    handleKnockdownState(p) {
        // Stay on ground until hit.
        // Apply strong friction
        if (this.isGrounded) {
            this.vx *= 0.5;
        }
    }

    setState(newState) {
        this.state = newState;
        this.stateTimer = 0;
        if (newState === "ATTACK") {
            this.hasHit = false;
        } else {
            this.currentAttackDirection = null; // Reset attack direction when not attacking
        }
    }

    takeDamage(amount, kbX, kbY) {
        if (this.state === "DEAD") return;
        this.health -= amount;
        
        // Logic for Knockdown
        if (this.state === "KNOCKDOWN") {
            // Hit while down -> Wake up
            applyKnockback(this, kbX, kbY);
            this.setState("STUN");
            this.comboHitCount = 0; // Reset combo to prevent loop
        } else {
            // Normal hit
            this.comboHitCount++;
            
            if (this.comboHitCount >= 2) {
                // Trigger Knockdown
                this.setState("KNOCKDOWN");
                applyKnockback(this, kbX, kbY); 
            } else {
                // Normal Stun
                applyKnockback(this, kbX, kbY);
                this.setState("STUN");
            }
        }
        
        // Hitstop effect
        gameState.hitStop = 5;
        gameState.camera.shake = 10;
        
        createExplosion(this.x, this.y, 10, 'SPARK');
        createExplosion(this.x, this.y, 5, 'BLOOD');
    }

    // Render Stick Figure
    render(p) {
        p.push();
        p.translate(this.x, this.y);
        if (this.facing === -1) p.scale(-1, 1);

        // Flicker if damaged recently?
        if (this.state === "STUN" && p.frameCount % 4 === 0) {
            p.pop();
            return; 
        }

        p.stroke(this.isPlayer ? 0 : 255); // Black for player, White for enemy (or configurable)
        
        if (this.isPlayer) {
            p.stroke(50, 255, 100); // Green tint for player
        } else {
            // Use custom enemy color
            p.stroke(this.color);
        }

        p.strokeWeight(3);
        p.noFill();

        // Handle Knockdown rotation
        if (this.state === "KNOCKDOWN") {
            p.translate(0, 15); // Adjust position to look like on ground
            p.rotate(-p.PI / 2); // Lay down
        }

        // Animation procedural logic
        let legOffset = 0;
        let armOffset = 0;
        
        if (this.state === "RUN" && this.isGrounded) {
            legOffset = Math.sin(p.frameCount * 0.5) * 10;
        }
        
        // Head
        p.circle(0, -20, 15);
        
        // Body
        p.line(0, -12, 0, 10);
        
        // Legs
        p.line(0, 10, -10 + legOffset, 30);
        p.line(0, 10, 10 - legOffset, 30);
        
        // Arms
        if (this.state === "ATTACK") {
            // Directional attack animations
            switch (this.currentAttackDirection) {
                case 'UP':
                    p.line(0, -5, 0, -30); // Arms up
                    break;
                case 'DOWN':
                    p.line(0, 5, 0, 30); // Arms down
                    break;
                case 'BACK':
                    p.line(0, -5, -20, -10); // Arm back
                    break;
                case 'FORWARD':
                case 'NEUTRAL':
                default:
                    p.line(0, -5, 20, -5); // Arm forward (punch)
                    break;
            }
        } else {
            p.line(0, -5, 15 + armOffset, 10);
            p.line(0, -5, -15 - armOffset, 10);
        }

        // Add visual feedback for attacks
        if (this.state === "ATTACK" && this.stateTimer < 10) { // Visual for first 10 frames of attack
            p.push();
            let alpha = p.map(this.stateTimer, 0, 10, 255, 0);
            p.stroke(255, 200, 50, alpha); // Yellowish attack glow
            p.strokeWeight(4);
            p.noFill();

            switch (this.currentAttackDirection) {
                case 'UP':
                    p.line(0, -20, 0, -50); // Line above head
                    p.arc(0, -50, 20, 20, p.PI, p.TWO_PI); // Small arc at top
                    break;
                case 'DOWN':
                    p.line(0, 20, 0, 50); // Line below feet
                    p.arc(0, 50, 20, 20, 0, p.PI); // Small arc at bottom
                    break;
                case 'BACK':
                    p.line(-10, -5, -40, -5); // Line behind
                    break;
                case 'FORWARD':
                case 'NEUTRAL':
                default:
                    p.line(10, -5, 40, -5); // Line in front (punch)
                    break;
            }
            p.pop();
        }

        // Debug Hitbox
        // p.stroke(0, 255, 0);
        // p.rect(0, 0, this.width, this.height);

        p.pop();
    }
}

// --- Player Class ---
export class Player extends Fighter {
    constructor(x, y) {
        super(x, y, true);
    }

    update(p) {
        // Imports
        const { isKeyDown, isKeyPressed, KEYS, consumeKeyPress } =  window.inputModule; // We will attach input to window for easy access in classes
        
        super.update(p);
        
        if (this.state === "DEAD") return;
        if (this.state === "KNOCKDOWN") return; // Block Input
        if (this.state === "STUN") {
            if (this.stateTimer > 20) {
                this.setState("IDLE"); // Recovery
                this.comboHitCount = 0; // Reset combo
            }
            return;
        }

        // Movement
        if (this.state !== "DASH" && this.state !== "ATTACK") {
            if (isKeyDown(KEYS.LEFT)) {
                this.vx = -this.moveSpeed;
                this.facing = -1;
                if (this.isGrounded) this.setState("RUN");
            } else if (isKeyDown(KEYS.RIGHT)) {
                this.vx = this.moveSpeed;
                this.facing = 1;
                if (this.isGrounded) this.setState("RUN");
            } else {
                this.vx = 0;
                if (this.isGrounded) this.setState("IDLE");
            }

            // Jump
            if (isKeyPressed(KEYS.SPACE)) {
                if (this.isGrounded) {
                    this.vy = this.jumpForce;
                    this.isGrounded = false;
                    this.setState("JUMP");
                    createExplosion(this.x, this.y + 30, 5, 'DUST');
                    consumeKeyPress(KEYS.SPACE);
                } else if (this.hasDoubleJump) {
                    this.vy = this.jumpForce;
                    this.hasDoubleJump = false;
                    createExplosion(this.x, this.y + 30, 5, 'DUST');
                    consumeKeyPress(KEYS.SPACE);
                }
            }
        }
        
        // Dash
        if (isKeyPressed(KEYS.SHIFT) && this.canDash && this.dashCooldown === 0) {
            // Can dash if idle/move/jump OR if Attacking and HasHit (Dash Cancel)
            if (this.state !== "ATTACK" || (this.state === "ATTACK" && this.hasHit)) {
                this.setState("DASH");
                this.canDash = false; // Need to land to reset or cooldown
                this.dashCooldown = 60;
                
                // Dash direction
                let dirX = 0;
                let dirY = 0;
                if (isKeyDown(KEYS.LEFT)) dirX = -1;
                if (isKeyDown(KEYS.RIGHT)) dirX = 1;
                if (isKeyDown(KEYS.UP)) dirY = -1;
                if (isKeyDown(KEYS.DOWN)) dirY = 1;
                
                if (dirX === 0 && dirY === 0) dirX = this.facing;
                
                // Normalize roughly
                const mag = Math.sqrt(dirX*dirX + dirY*dirY);
                if (mag > 0) { // Avoid division by zero
                    this.vx = (dirX / mag) * this.dashSpeed;
                    this.vy = (dirY / mag) * this.dashSpeed;
                } else { // Default to forward dash if no direction
                    this.vx = this.facing * this.dashSpeed;
                    this.vy = 0;
                }
                
                consumeKeyPress(KEYS.SHIFT);
                createExplosion(this.x, this.y, 8, 'LINE');
            }
        }

        // Attack
        if (isKeyPressed(KEYS.Z) && this.attackCooldown === 0 && this.state !== "DASH") {
            this.setState("ATTACK");
            this.attackCooldown = 20;
            
            let hbX, hbY, hbWidth, hbHeight, hbDamage, hbKbX, hbKbY;

            let attackDirX = 0;
            let attackDirY = 0;
            if (isKeyDown(KEYS.UP)) attackDirY = -1;
            if (isKeyDown(KEYS.DOWN)) attackDirY = 1;
            if (isKeyDown(KEYS.LEFT)) attackDirX = -1;
            if (isKeyDown(KEYS.RIGHT)) attackDirX = 1;

            // Determine attack type based on directional input
            if (attackDirY === -1) { // Up Attack (Prioritize vertical)
                this.currentAttackDirection = 'UP';
                hbX = this.x;
                hbY = this.y - (this.height / 2) - 20; // Above head
                hbWidth = 40;
                hbHeight = 40;
                hbDamage = 12;
                hbKbX = 0;
                hbKbY = -15; // Strong vertical knockback
            } else if (attackDirY === 1) { // Down Attack (Prioritize vertical)
                this.currentAttackDirection = 'DOWN';
                hbX = this.x;
                hbY = this.y + (this.height / 2) + 20; // Below feet
                hbWidth = 40;
                hbHeight = 40;
                hbDamage = 8;
                hbKbX = 0;
                hbKbY = 10; // Downward knockback (spike-like)
            } else if ((attackDirX === -1 && this.facing === 1) || (attackDirX === 1 && this.facing === -1)) { 
                // Back Attack: Attacking opposite to current facing direction
                this.currentAttackDirection = 'BACK';
                hbX = this.x + (30 * -this.facing); // Behind player
                hbY = this.y;
                hbWidth = 40;
                hbHeight = 40;
                hbDamage = 8;
                hbKbX = 8 * -this.facing;
                hbKbY = -3; // Weaker, for spacing/repositioning
            } else if ((attackDirX === 1 && this.facing === 1) || (attackDirX === -1 && this.facing === -1)) { 
                // Explicit Forward Attack: Attacking in current facing direction
                this.currentAttackDirection = 'FORWARD';
                hbX = this.x + (30 * this.facing); // In front of player
                hbY = this.y;
                hbWidth = 40;
                hbHeight = 40;
                hbDamage = 10;
                hbKbX = 10 * this.facing;
                hbKbY = -5;
            } else { // Neutral Attack (Default if no specific direction or conflicting horizontal)
                this.currentAttackDirection = 'NEUTRAL';
                hbX = this.x + (30 * this.facing); // In front of player
                hbY = this.y;
                hbWidth = 40;
                hbHeight = 40;
                hbDamage = 10;
                hbKbX = 10 * this.facing;
                hbKbY = -5;
            }
            
            const hb = new Hitbox(this, hbX, hbY, hbWidth, hbHeight, hbDamage, hbKbX, hbKbY, 10);
            gameState.projectiles.push(hb);
            consumeKeyPress(KEYS.Z);
        }
        
        if (this.state === "ATTACK") {
            // Apply a little friction during ground attack
            if (this.isGrounded) this.vx *= 0.5;
            if (this.stateTimer > 15) {
                this.setState("IDLE");
            }
        }
        
        if (this.state === "DASH") {
            this.vy = 0; // defy gravity
            if (this.stateTimer > 10) {
                this.vx *= 0.2;
                this.vy *= 0.2;
                this.setState("FALL");
            }
        }
    }
}

// --- Enemy AI ---
export class Enemy extends Fighter {
    constructor(x, y, level = 1, color = [255, 50, 50]) {
        super(x, y, false);
        this.decisionTimer = 0;
        this.target = null;
        this.level = level;
        this.color = color;
        
        // Scale stats based on level
        this.moveSpeed = 5 + (level * 0.5);
        this.damageMultiplier = 1 + (level * 0.15);
        this.maxHealth = 100 + (level * 10);
        this.health = this.maxHealth;
    }

    update(p) {
        super.update(p);
        this.target = gameState.player;
        if (!this.target) return;
        
        if (this.state === "DEAD" || this.state === "STUN" || this.state === "KNOCKDOWN") return;
        
        const dist = Math.abs(this.target.x - this.x);
        const dy = this.target.y - this.y;
        
        // Simple AI
        this.decisionTimer++;
        if (this.decisionTimer > 30) { // React every half second roughly
             this.decisionTimer = 0;
        }

        // Face player
        if (this.target.x > this.x) this.facing = 1;
        else this.facing = -1;

        if (this.state !== "ATTACK" && this.state !== "DASH") {
            // Movement
            if (dist > 50) {
                this.vx = this.moveSpeed * this.facing;
                this.setState("RUN");
            } else {
                this.vx = 0;
                this.setState("IDLE");
                
                // Attack if close
                if (this.attackCooldown === 0 && Math.random() < 0.1) {
                     this.setState("ATTACK");
                     this.attackCooldown = 40; // Slower than player
                     const hbX = this.x + (30 * this.facing);
                     const damage = 8 * this.damageMultiplier;
                     const hb = new Hitbox(this, hbX, this.y, 40, 40, damage, 8 * this.facing, -5, 10);
                     gameState.projectiles.push(hb);
                }
            }
            
            // Jump if target is above or random
            if ((dy < -50 || Math.random() < 0.01) && this.isGrounded) {
                this.vy = this.jumpForce;
                this.isGrounded = false;
                this.setState("JUMP");
            }
        }
        
        if (this.state === "ATTACK") {
             if (this.stateTimer > 20) this.setState("IDLE");
        }
    }
}
