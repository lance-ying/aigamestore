<analysis>
- Issue identified: The fishing bobber launch velocity was calculated based on values too high for the frame-based physics update loop, causing it to fly extremely far (teleporting across the map). Additionally, the fishing mini-game parameters (target size, drain rate, fill rate) were tuned too difficult for the user.
- Files affected: `fishing.js`, `ui.js`
- Changes needed: 
    - In `fishing.js`: Drastically reduce the velocity multipliers in `performCast` to suit the physics scale. Adjust `updateReelingGame` to have a larger target zone, larger player bar, slower progress drain, and faster progress fill. Move imports to the top of the file to fix syntax validity.
    - In `ui.js`: Update the visual rendering of the fishing mini-game bars (target zone and player bar) to match the new, larger logical sizes.
</analysis>

import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';
import { gameState, FISH_TYPES, logGameInfo } from './globals.js';
import { input } from './input.js';
import { PhysicsBody } from './physics.js';

export class FishingSystem {
    constructor(player) {
        this.player = player;
        this.state = 'IDLE'; // IDLE, CASTING, WAITING, BITING, HOOKED, REELING, CAUGHT_ANIM
        this.castPower = 0;
        this.bobber = null;
        this.line = null;
        
        // Mini-game vars
        this.biteTimer = 0;
        this.hookTimer = 0;
        this.targetPos = 0.5; // 0 to 1
        this.playerBarPos = 0.5; // 0 to 1
        this.progress = 0.3; // 0 to 1, start partial
        this.targetVelocity = 0.01;
        this.barVelocity = 0;
        this.currentFish = null;
        
        this.animTimer = 0;
        this.lastCaughtFish = null;
    }
    
    get isActive() {
        return this.state !== 'IDLE';
    }
    
    update(deltaTime) {
        // Update Fishing Line Visual
        if (this.line && this.bobber && this.player) {
            const positions = this.line.geometry.attributes.position.array;
            
            const start = this.player.getRodTipWorldPosition();
            const end = this.bobber.mesh.position;
            
            positions[0] = start.x;
            positions[1] = start.y;
            positions[2] = start.z;
            positions[3] = end.x;
            positions[4] = end.y;
            positions[5] = end.z;
            
            this.line.geometry.attributes.position.needsUpdate = true;
        }

        switch(this.state) {
            case 'IDLE':
                // Handled by input outside
                break;
                
            case 'CASTING':
                // Charge up power
                if (input.isDown('ACTION')) {
                    this.castPower += deltaTime * 1.5; // fill in ~0.6s
                    if (this.castPower > 1) this.castPower = 1;
                } else {
                    // Released
                    this.performCast();
                }
                break;
                
            case 'WAITING':
                // Bobber physics update happens in main entity loop, 
                // but we check time here.
                this.biteTimer -= deltaTime;
                if (this.biteTimer <= 0) {
                    this.state = 'BITING';
                    this.hookTimer = 1.0; // 1 second to react
                    // SFX or Visual Shake could trigger here
                    if(this.bobber) this.bobber.dip();
                }
                // Cancel if player moves too much? Nah, allow it for now.
                break;
                
            case 'BITING':
                this.hookTimer -= deltaTime;
                if (input.justPressed('ACTION')) {
                    this.startReeling();
                } else if (this.hookTimer <= 0) {
                    this.failFishing("Too slow!");
                }
                break;
                
            case 'REELING':
                this.updateReelingGame(deltaTime);
                break;
                
            case 'CAUGHT_ANIM':
                this.animTimer -= deltaTime;
                if (this.animTimer <= 0) {
                    this.state = 'IDLE';
                    this.cleanupBobber();
                }
                break;
        }
    }
    
    startCasting() {
        if (!this.player) return;
        this.state = 'CASTING';
        this.castPower = 0;
    }
    
    performCast() {
        this.state = 'WAITING';
        this.biteTimer = 2.0 + Math.random() * 3.0; // 2-5s wait
        
        // Spawn Bobber
        const forward = new THREE.Vector3(0, 0, 1).applyQuaternion(this.player.mesh.quaternion);
        const startPos = this.player.getRodTipWorldPosition();
        
        // Calculate velocity based on power (Reduced for easier control/physics scale)
        const velocity = forward.multiplyScalar(0.4 + this.castPower * 0.6);
        velocity.y = 0.2 + this.castPower * 0.4; // Arc
        
        this.bobber = new Bobber(startPos, velocity);
        gameState.entities.push(this.bobber);
        gameState.scene.add(this.bobber.mesh);
        
        // Create Line
        const lineGeo = new THREE.BufferGeometry();
        const vertices = new Float32Array(6); // 2 points * 3 coords
        lineGeo.setAttribute('position', new THREE.BufferAttribute(vertices, 3));
        const lineMat = new THREE.LineBasicMaterial({ color: 0xFFFFFF, linewidth: 2 });
        this.line = new THREE.Line(lineGeo, lineMat);
        gameState.scene.add(this.line);
        
        // Select potential fish
        this.currentFish = this.selectRandomFish();
    }
    
    selectRandomFish() {
        // Weighted random
        const roll = Math.random() * 10;
        const fish = FISH_TYPES[Math.floor(Math.random() * FISH_TYPES.length)];
        // Simple selection for now
        return fish;
    }
    
    startReeling() {
        this.state = 'REELING';
        this.progress = 0.3;
        this.playerBarPos = 0.5;
        this.targetPos = 0.5;
        // Difficulty scaling - Reduced base speed for easier gameplay
        this.targetVelocity = 0.002 * (this.currentFish ? this.currentFish.difficulty : 1);
    }
    
    updateReelingGame(deltaTime) {
        // Player Input - Force
        const force = 8.0; 
        
        if (input.isDown('ACTION')) {
            this.barVelocity += deltaTime * force;
        } else {
            this.barVelocity -= deltaTime * force;
        }
        
        // Physics for bar
        this.barVelocity *= 0.9; // damping
        this.playerBarPos += this.barVelocity * deltaTime;
        
        // Clamp
        if (this.playerBarPos < 0) { this.playerBarPos = 0; this.barVelocity = 0; }
        if (this.playerBarPos > 1) { this.playerBarPos = 1; this.barVelocity = 0; }
        
        // Target Movement (Fish AI)
        // Moves sinusoidally plus noise
        const time = Date.now() / 1000;
        const difficulty = this.currentFish.difficulty;
        
        // Erratic movement
        if (Math.random() < 0.05 * difficulty) {
            this.targetVelocity = (Math.random() - 0.5) * 1.5 * difficulty; // Reduced erratic speed
        }
        
        this.targetPos += this.targetVelocity * deltaTime;
        
        // Keep target in bounds but bounce
        if (this.targetPos < 0) { this.targetPos = 0; this.targetVelocity *= -1; }
        if (this.targetPos > 1) { this.targetPos = 1; this.targetVelocity *= -1; }
        
        // Check Overlap (Hit testing)
        // Increased sizes for easier gameplay
        const barSize = 0.15; // Larger player bar
        const targetSize = 0.35; // Larger target zone (very generous)
        
        // We want player bar (point) inside target zone
        const dist = Math.abs(this.playerBarPos - this.targetPos);
        const inZone = dist < (targetSize / 2);
        
        if (inZone) {
            this.progress += deltaTime * 0.4; // Fill bar faster
        } else {
            this.progress -= deltaTime * 0.05; // Drain bar much slower
        }
        
        // Win/Loss
        if (this.progress >= 1.0) {
            this.successCatch();
        } else if (this.progress <= 0) {
            this.failFishing("Fish got away...");
        }
    }
    
    successCatch() {
        this.state = 'CAUGHT_ANIM';
        this.animTimer = 2.0;
        this.lastCaughtFish = this.currentFish;
        
        gameState.inventory.push(this.currentFish);
        gameState.money += this.currentFish.value;
        gameState.score += this.currentFish.value;
        
        gameState.chatLog.push({
            sender: "System",
            message: `You caught a ${this.currentFish.name}!`,
            color: '#FFFF00'
        });
        
        // Check win
        if (gameState.inventory.length >= gameState.goalFishCount) {
            setTimeout(() => {
                gameState.gamePhase = "GAME_OVER_WIN";
            }, 2000);
        }
    }
    
    failFishing(reason) {
        this.state = 'IDLE';
        this.cleanupBobber();
        gameState.chatLog.push({
            sender: "System",
            message: reason,
            color: '#FF9999'
        });
    }
    
    cleanupBobber() {
        if (this.bobber) {
            gameState.scene.remove(this.bobber.mesh);
            gameState.entities = gameState.entities.filter(e => e !== this.bobber);
            this.bobber = null;
        }
        if (this.line) {
            gameState.scene.remove(this.line);
            this.line = null;
        }
    }
}

// Bobber Entity
class Bobber extends PhysicsBody {
    constructor(position, velocity) {
        const geometry = new THREE.SphereGeometry(0.2, 8, 8);
        const material = new THREE.MeshLambertMaterial({ color: 0xFF0000 });
        const mesh = new THREE.Mesh(geometry, material);
        mesh.position.copy(position);
        
        super(mesh, { mass: 0.1, drag: 0.01, radius: 0.2 });
        this.velocity.copy(velocity);
        
        this.startY = position.y;
    }
    
    dip() {
        // Visual splash effect
        this.velocity.y = -2.0;
        // Could spawn particle
    }
}

import { gameState, COLORS, CANVAS_WIDTH, CANVAS_HEIGHT } from './globals.js';

class UIManager {
    constructor() {
        this.canvas = document.createElement('canvas');
        this.canvas.width = CANVAS_WIDTH;
        this.canvas.height = CANVAS_HEIGHT;
        this.canvas.style.position = 'absolute';
        this.canvas.style.top = '0';
        this.canvas.style.left = '0';
        this.canvas.style.pointerEvents = 'none'; // Click through
        this.ctx = this.canvas.getContext('2d');
    }

    init() {
        if (gameState.gameContainer) {
            gameState.gameContainer.appendChild(this.canvas);
        }
    }

    render() {
        this.ctx.clearRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);

        if (gameState.gamePhase === "START") {
            this.renderStartScreen();
        } else if (gameState.gamePhase === "PLAYING") {
            this.renderHUD();
            this.renderFishingUI();
        } else if (gameState.gamePhase === "PAUSED") {
            // No visual overlay or text for pause as per instructions
        } else if (gameState.gamePhase.startsWith("GAME_OVER")) {
            this.renderGameOver();
        }
    }

    renderStartScreen() {
        this.ctx.fillStyle = 'rgba(0,0,0,0.8)';
        this.ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);
        
        // New title message
        this.ctx.fillStyle = 'white';
        this.ctx.font = 'bold 30px monospace'; // Use original title font size for prominence
        this.ctx.textAlign = 'center';
        this.ctx.fillText("press enter to begin", CANVAS_WIDTH/2, CANVAS_HEIGHT/2 - 50); // Centered where the title was
        
        // Keep controls section
        this.ctx.fillStyle = '#AAAAAA';
        this.ctx.font = '14px monospace';
        this.ctx.fillText("Arrows/WASD to Move | Z to Fish | SPACE to Jump", CANVAS_WIDTH/2, CANVAS_HEIGHT/2 + 50); // Positioned below the new title
    }

    renderHUD() {
        // Top Left: Money & Journal
        this.ctx.fillStyle = 'rgba(0,0,0,0.5)';
        this.ctx.roundRect ? this.ctx.roundRect(10, 10, 150, 80, 5) : this.ctx.fillRect(10, 10, 150, 80);
        this.ctx.fill();
        
        this.ctx.fillStyle = 'gold';
        this.ctx.font = '16px monospace';
        this.ctx.textAlign = 'left';
        this.ctx.fillText(`$ ${gameState.money}`, 20, 35);
        
        this.ctx.fillStyle = 'white';
        this.ctx.fillText(`Fish: ${gameState.inventory.length}/${gameState.goalFishCount}`, 20, 55);
        
        this.ctx.fillStyle = 'lightgreen';
        this.ctx.fillText(`Score: ${gameState.score}`, 20, 75);

        // Bottom Left: Chat Log
        this.renderChat();
    }

    renderChat() {
        const x = 10;
        const y = CANVAS_HEIGHT - 120;
        const w = 250;
        const h = 110;
        
        this.ctx.fillStyle = 'rgba(0,0,0,0.3)';
        this.ctx.fillRect(x, y, w, h);
        
        this.ctx.font = '12px monospace';
        this.ctx.textAlign = 'left';
        
        // Show last 5 messages
        const messages = gameState.chatLog.slice(-5);
        messages.forEach((msg, index) => {
            this.ctx.fillStyle = msg.color || 'white';
            this.ctx.fillText(`${msg.sender}: ${msg.message}`, x + 5, y + 20 + (index * 18));
        });
    }

    renderFishingUI() {
        if (!gameState.fishingSystem || !gameState.fishingSystem.isActive) return;

        const fs = gameState.fishingSystem;
        const cx = CANVAS_WIDTH / 2;
        const cy = CANVAS_HEIGHT / 2;

        // Visual for Casting Power
        if (fs.state === 'CASTING') {
            this.ctx.fillStyle = 'white';
            this.ctx.font = '16px monospace';
            this.ctx.textAlign = 'center';
            this.ctx.fillText("HOLD Z TO CHARGE", cx, cy - 50);
            
            // Power Bar
            this.ctx.fillStyle = 'black';
            this.ctx.fillRect(cx - 50, cy - 30, 100, 10);
            this.ctx.fillStyle = 'lime';
            this.ctx.fillRect(cx - 50, cy - 30, 100 * fs.castPower, 10);
        }

        // Waiting
        if (fs.state === 'WAITING') {
             this.ctx.fillStyle = 'white';
             this.ctx.font = 'italic 14px monospace';
             this.ctx.textAlign = 'center';
             this.ctx.fillText("Waiting for bite...", cx, cy - 50);
        }

        // Biting Alert
        if (fs.state === 'BITING') {
            this.ctx.fillStyle = 'red';
            this.ctx.font = 'bold 32px monospace';
            this.ctx.textAlign = 'center';
            this.ctx.fillText("!!!", cx, cy - 80);
            this.ctx.font = '16px monospace';
            this.ctx.fillText("PRESS Z!", cx, cy - 50);
        }

        // Reeling Mini-game
        if (fs.state === 'REELING') {
            // Draw background bar (vertical)
            const bh = 150;
            const bw = 20;
            const bx = cx + 60;
            const by = cy - bh/2;
            
            this.ctx.fillStyle = 'rgba(0,0,0,0.8)';
            this.ctx.fillRect(bx, by, bw, bh);
            
            // Target Zone (The fish's comfort zone) - Moves
            // Increased visual size to match logic (0.35 * 150 approx 50)
            const th = 50;

            this.ctx.fillRect(bx, ty, bw, th);

            const ph = 20;

            this.ctx.textAlign = 'center';
            this.ctx.fillText("KEEP THE BAR ON THE GREEN!", cx, cy + 80);

    renderGameOver() {
        this.ctx.fillStyle = 'rgba(0,0,0,0.9)';
        this.ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);

        this.ctx.font = '16px monospace';
        this.ctx.fillText("Press R to Play Again", CANVAS_WIDTH/2, 320);
