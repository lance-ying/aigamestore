<analysis>
- Issue identified: After careful analysis, I believe the issue is in the vehicle physics update loop. The `isAccelerating`, `isBraking`, and `isDrifting` flags are being reset to `false` at the END of the `update()` method in entities.js. However, the `isDrifting` flag is being set in `handleInput()` through the handbrake check, but the RESET at the end of `update()` happens BEFORE the next frame's input handling. This means that even if the player releases the space bar, `isDrifting` might still be true from the previous frame during the velocity calculation, causing excessive friction. Additionally, the turn methods `turnLeft()` and `turnRight()` accumulate angular velocity, which should also be reset properly each frame.

- Files affected: entities.js, input.js

- Changes needed: 
  1. Move the `isDrifting` reset to happen in `handleInput()` BEFORE applying input, not in the update method
  2. Ensure angular velocity doesn't accumulate from turn methods  by resetting it each frame BEFORE applying new turn input
  3. Increase acceleration power slightly more for better responsiveness

</analysis>

// entities.js - Game entity classes
import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';
import { gameState, logPlayerInfo } from './globals.js';

export class Vehicle {
  constructor(x, y, z) {
    // Create vehicle mesh (sports car style)
    const group = new THREE.Group();
    
    // Car body
    const bodyGeometry = new THREE.BoxGeometry(2, 0.8, 4);
    const bodyMaterial = new THREE.MeshStandardMaterial({
      color: 0xFF4500,
      metalness: 0.7,
      roughness: 0.3
    });
    const body = new THREE.Mesh(bodyGeometry, bodyMaterial);
    body.position.y = 0.6;
    body.castShadow = true;
    body.receiveShadow = true;
    group.add(body);
    
    // Car cabin
    const cabinGeometry = new THREE.BoxGeometry(1.6, 0.7, 2);
    const cabinMaterial = new THREE.MeshStandardMaterial({
      color: 0x1A1A1A,
      metalness: 0.5,
      roughness: 0.5
    });
    const cabin = new THREE.Mesh(cabinGeometry, cabinMaterial);
    cabin.position.y = 1.2;
    cabin.position.z = -0.3;
    cabin.castShadow = true;
    group.add(cabin);
    
    // Wheels
    const wheelGeometry = new THREE.CylinderGeometry(0.4, 0.4, 0.3, 16);
    const wheelMaterial = new THREE.MeshStandardMaterial({
      color: 0x1A1A1A,
      metalness: 0.3,
      roughness: 0.7
    });
    
    const wheelPositions = [
      [-1.1, 0.4, 1.2],  // Front left
      [1.1, 0.4, 1.2],   // Front right
      [-1.1, 0.4, -1.2], // Back left
      [1.1, 0.4, -1.2]   // Back right
    ];
    
    this.wheels = [];
    wheelPositions.forEach(pos => {
      const wheel = new THREE.Mesh(wheelGeometry, wheelMaterial);
      wheel.rotation.z = Math.PI / 2;
      wheel.position.set(...pos);
      wheel.castShadow = true;
      group.add(wheel);
      this.wheels.push(wheel);
    });
    
    // Headlights
    const lightGeometry = new THREE.SphereGeometry(0.15, 8, 8);
    const lightMaterial = new THREE.MeshStandardMaterial({
      color: 0xFFFFFF,
      emissive: 0xFFFF00,
      emissiveIntensity: 0.8
    });
    
    const leftLight = new THREE.Mesh(lightGeometry, lightMaterial);
    leftLight.position.set(-0.6, 0.5, 2.1);
    group.add(leftLight);
    
    const rightLight = new THREE.Mesh(lightGeometry, lightMaterial);
    rightLight.position.set(0.6, 0.5, 2.1);
    group.add(rightLight);
    
    this.mesh = group;
    this.mesh.position.set(x, y, z);
    gameState.scene.add(this.mesh);
    
    // Physics properties
    this.velocity = new THREE.Vector3(0, 0, 0);
    this.acceleration = new THREE.Vector3(0, 0, 0);
    this.angularVelocity = 0;
    
    // Vehicle properties - OPTIMIZED FOR BETTER RESPONSIVENESS
    this.maxSpeed = 1.2;
    this.accelerationPower = 0.2; // Increased from 0.15 to 0.2 for better acceleration
    this.brakePower = 0.06;
    this.turnSpeed = 0.04;
    this.driftFactor = 0.95;
    this.friction = 0.98; // Less resistance for smoother movement
    this.boostMultiplier = 2.0;
    
    // State
    this.currentSpeed = 0;
    this.isAccelerating = false;
    this.isBraking = false;
    this.isDrifting = false;
    this.isBoosting = false;
    this.boostTimer = 0;
    this.boostDuration = 2.0;
    
    // Collision
    this.radius = 2.5;
    this.lastPosition = new THREE.Vector3().copy(this.mesh.position);
    
    // Animation
    this.wheelRotation = 0;
  }
  
  update(deltaTime) {
    // Store last position for collision response
    this.lastPosition.copy(this.mesh.position);
    
    // Update boost timer
    if (this.isBoosting) {
      this.boostTimer -= deltaTime;
      if (this.boostTimer <= 0) {
        this.isBoosting = false;
      }
    }
    
    // Calculate forward direction
    const forward = new THREE.Vector3(0, 0, -1);
    forward.applyQuaternion(this.mesh.quaternion);
    
    // Apply acceleration
    if (this.isAccelerating) {
      const power = this.isBoosting ? 
        this.accelerationPower * this.boostMultiplier : 
        this.accelerationPower;
      this.acceleration.add(forward.clone().multiplyScalar(power));
    }
    
    // Apply braking
    if (this.isBraking) {
      this.velocity.multiplyScalar(1 - this.brakePower);
    }
    
    // Update velocity
    this.velocity.add(this.acceleration);
    
    // Apply friction
    const frictionAmount = this.isDrifting ? this.driftFactor : this.friction;
    this.velocity.multiplyScalar(frictionAmount);
    
    // Clamp speed
    const maxSpeed = this.isBoosting ? this.maxSpeed * this.boostMultiplier : this.maxSpeed;
    this.currentSpeed = this.velocity.length();
    if (this.currentSpeed > maxSpeed) {
      this.velocity.normalize().multiplyScalar(maxSpeed);
      this.currentSpeed = maxSpeed;
    }
    
    // Update position
    this.mesh.position.add(this.velocity.clone().multiplyScalar(deltaTime * 60));
    
    // Update rotation based on angular velocity
    // Allow rotation even when stationary (at reduced rate)
    if (Math.abs(this.angularVelocity) > 0.001) {
      // Use minimum turn factor of 0.3 when stationary, scale up with speed
      const speedFactor = this.currentSpeed > 0.05 ? 
        (this.currentSpeed / this.maxSpeed) : 0.3;
      const turnAmount = this.angularVelocity * speedFactor;
      this.mesh.rotateY(turnAmount);
    }
    
    // Keep vehicle grounded
    this.mesh.position.y = 0.2;
    
    // Animate wheels
    this.wheelRotation += this.currentSpeed * 0.5;
    this.wheels.forEach(wheel => {
      wheel.rotation.x = this.wheelRotation;
    });
    
    // Reset acceleration for next frame
    this.acceleration.set(0, 0, 0);
    this.isAccelerating = false;
    this.isBraking = false;
    
    // Log position periodically
    if (gameState.frameCount % 30 === 0) {
      this.logPosition();
    }
  }
  
  accelerate() {
    this.isAccelerating = true;
  }
  
  brake() {
    this.isBraking = true;
  }
  
  turnLeft() {
    this.angularVelocity += this.turnSpeed;
  }
  
  turnRight() {
    this.angularVelocity -= this.turnSpeed;
  }
  
  handbrake() {
    this.isDrifting = true;
    this.velocity.multiplyScalar(0.95);
  }
  
  activateBoost() {
    if (gameState.boostsAvailable > 0 && !this.isBoosting) {
      this.isBoosting = true;
      this.boostTimer = this.boostDuration;
      gameState.boostsAvailable--;
    }
  }
  
  handleCollision(obstacle) {
    // Move back to last position
    this.mesh.position.copy(this.lastPosition);
    
    // Reduce velocity significantly
    this.velocity.multiplyScalar(0.3);
    
    // Bounce back slightly
    const direction = new THREE.Vector3()
      .subVectors(this.mesh.position, obstacle.position)
      .normalize();
    this.velocity.add(direction.multiplyScalar(0.2));
  }
  
  logPosition() {
    const screenPos = this.mesh.position.clone().project(gameState.camera);
    logPlayerInfo(
      (screenPos.x + 1) * 300,
      (1 - screenPos.y) * 200,
      this.mesh.position.x,
      this.mesh.position.y,
      this.mesh.position.z,
      {
        speed: this.currentSpeed,
        isBoosting: this.isBoosting,
        tokensCollected: gameState.tokensCollected
      }
    );
  }
}

export class Token {
  constructor(x, y, z) {
    // Create token mesh (glowing ring)
    const geometry = new THREE.TorusGeometry(0.5, 0.15, 16, 32);
    const material = new THREE.MeshStandardMaterial({
      color: 0xFFD700,
      emissive: 0xFFD700,
      emissiveIntensity: 0.8,
      metalness: 0.7,
      roughness: 0.3
    });
    
    this.mesh = new THREE.Mesh(geometry, material);
    this.mesh.position.set(x, y, z);
    this.mesh.castShadow = true;
    
    // Outer glow ring
    const glowGeometry = new THREE.TorusGeometry(0.7, 0.1, 16, 32);
    const glowMaterial = new THREE.MeshBasicMaterial({
      color: 0xFFFF00,
      transparent: true,
      opacity: 0.3
    });
    this.glow = new THREE.Mesh(glowGeometry, glowMaterial);
    this.glow.position.copy(this.mesh.position);
    
    gameState.scene.add(this.mesh);
    gameState.scene.add(this.glow);
    
    this.value = 10;
    this.rotationSpeed = 0.03;
    this.bobSpeed = 0.002;
    this.bobAmount = 0.3;
    this.initialY = y;
    this.collected = false;
  }
  
  update(deltaTime) {
    if (this.collected) return;
    
    // Rotate
    this.mesh.rotation.y += this.rotationSpeed;
    this.glow.rotation.y -= this.rotationSpeed * 0.5;
    
    // Bob up and down
    const bobOffset = Math.sin(gameState.frameCount * this.bobSpeed) * this.bobAmount;
    this.mesh.position.y = this.initialY + bobOffset;
    this.glow.position.y = this.mesh.position.y;
    
    // Pulse glow
    this.glow.scale.setScalar(1 + Math.sin(gameState.frameCount * 0.05) * 0.1);
    
    // Check collision with player
    if (gameState.player) {
      const distance = this.mesh.position.distanceTo(gameState.player.mesh.position);
      if (distance < 2.0) {
        this.collect();
      }
    }
  }
  
  collect() {
    this.collected = true;
    gameState.score += this.value;
    gameState.tokensCollected++;
    
    // Add boost charge every few tokens
    if (gameState.tokensCollected % 3 === 0) {
      gameState.boostsAvailable++;
    }
    
    // Remove from scene
    gameState.scene.remove(this.mesh);
    gameState.scene.remove(this.glow);
    
    // Remove from tokens array
    const index = gameState.tokens.indexOf(this);
    if (index > -1) {
      gameState.tokens.splice(index, 1);
    }
    
    // Spawn new token
    spawnToken();
  }
}

      color: 0x00FF00,
      transparent: true,
      opacity: 0.2,
      side: THREE.DoubleSide
    });

    this.plane.position.y = 4;

    this.index = index;
    this.collected = false;
    this.value = 50;
    this.pulsePhase = 0;
  }
  
  update(deltaTime) {
    if (this.collected) return;
    
    // Pulse effect
    this.pulsePhase += 0.05;
    const scale = 1 + Math.sin(this.pulsePhase) * 0.1;
    this.mesh.scale.set(scale, 1, scale);
    
    // Update material emissive intensity
    this.mesh.children.forEach(child => {
      if (child.material && child.material.emissive) {
        child.material.emissiveIntensity = 0.3 + Math.sin(this.pulsePhase) * 0.3;
      }
    });
    
    // Check collision with player
    if (gameState.player) {
      const distance = this.mesh.position.distanceTo(gameState.player.mesh.position);
      if (distance < 4.0) {
        this.collect();
      }
    }
  }
  
  collect() {
    this.collected = true;
    gameState.score += this.value;
    gameState.checkpointsCompleted++;
    
    // Remove from scene
    gameState.scene.remove(this.mesh);
    
    // Spawn next checkpoint
    spawnCheckpoint();
  }
}

export function spawnToken() {
  const x = (Math.random() - 0.5) * 180;
  const z = (Math.random() - 0.5) * 180;
  const y = 2;
  
  const token = new Token(x, y, z);
  gameState.tokens.push(token);
  gameState.entities.push(token);
}

export function spawnCheckpoint() {
  // Random position away from player
  let x, z;
  let attempts = 0;
  
  do {
    x = (Math.random() - 0.5) * 160;
    z = (Math.random() - 0.5) * 160;
    attempts++;
  } while (
    gameState.player && 
    Math.sqrt(x * x + z * z) < 30 && 
    attempts < 10
  );
  
  const checkpoint = new Checkpoint(x, 0, z, gameState.checkpointIndex++);
  gameState.checkpoints.push(checkpoint);
  gameState.entities.push(checkpoint);
  gameState.activeCheckpoint = checkpoint;
}

  // Spawn initial tokens

    spawnToken();
  }

  spawnCheckpoint();
}

// input.js - Input handling and control modes
import { gameState, logInput, logGameInfo } from './globals.js';

const keys = {};

export function setupInput() {
  // Keydown handler
  document.addEventListener('keydown', (event) => {
    // Prevent default for game keys
    if ([13, 27, 32, 37, 38, 39, 40, 65, 68, 82, 83, 87, 16].includes(event.keyCode)) {
      event.preventDefault();
    }
    
    logInput('keydown', event.key, event.keyCode);
    keys[event.keyCode] = true;
    
    handlePhaseControls(event.keyCode);
  });
  
  // Keyup handler
  document.addEventListener('keyup', (event) => {
    logInput('keyup', event.key, event.keyCode);
    keys[event.keyCode] = false;
  });
}

function handlePhaseControls(keyCode) {
  // ENTER - Start game
  if (keyCode === 13 && gameState.gamePhase === "START") {
    gameState.gamePhase = "PLAYING";
    logGameInfo("PLAYING");
  }
  
  // ESC - Pause/Resume
  if (keyCode === 27) {
    if (gameState.gamePhase === "PLAYING") {
      gameState.gamePhase = "PAUSED";
      logGameInfo("PAUSED");
    } else if (gameState.gamePhase === "PAUSED") {
      gameState.gamePhase = "PLAYING";
      logGameInfo("PLAYING");
    }
  }
  
  // R - Restart
  if (keyCode === 82) {
    if (gameState.gamePhase === "GAME_OVER_WIN" || 
        gameState.gamePhase === "GAME_OVER_LOSE" ||
        gameState.gamePhase === "PAUSED") {
      resetGame();
    }
  }
}

export function handleInput() {
  if (gameState.gamePhase !== "PLAYING" || !gameState.player) return;
  
  const player = gameState.player;
  
  // Reset per-frame state flags BEFORE applying input
  player.isDrifting = false;
  player.angularVelocity *= 0.85; // Reduce angular velocity each frame
  
  // Use control mode
  if (gameState.controlMode === "HUMAN") {
    handleHumanInput(player);
  } else if (gameState.controlMode.startsWith("TEST_")) {
    handleTestInput(player);
  }
}

function handleHumanInput(player) {
  // Forward movement (W or Up Arrow)
  if (keys[87] || keys[38]) {
    player.accelerate();
  }
  
  // Backward movement (S or Down Arrow)
  if (keys[83] || keys[40]) {
    player.brake();
  }
  
  // Turn left (A or Left Arrow)
  if (keys[65] || keys[37]) {
    player.turnLeft();
  }
  
  // Turn right (D or Right Arrow)
  if (keys[68] || keys[39]) {
    player.turnRight();
  }
  
  // Handbrake (Space)
  if (keys[32]) {
    player.handbrake();
  }
  
  // Boost (Shift)
  if (keys[16]) {
    player.activateBoost();
  }
}

function resetGame() {
  // Clear entities
  gameState.tokens.forEach(token => {
    gameState.scene.remove(token.mesh);
    gameState.scene.remove(token.glow);
  });
  gameState.tokens = [];
  
  gameState.checkpoints.forEach(checkpoint => {
    gameState.scene.remove(checkpoint.mesh);
  });
  gameState.checkpoints = [];
  gameState.activeCheckpoint = null;
  gameState.checkpointIndex = 0;
  
  // Reset player
  if (gameState.player) {
    gameState.player.mesh.position.set(0, 0.2, 0);
    gameState.player.mesh.rotation.set(0, 0, 0);
    gameState.player.velocity.set(0, 0, 0);
    gameState.player.angularVelocity = 0;
    gameState.player.currentSpeed = 0;
    gameState.player.isBoosting = false;
    gameState.player.boostTimer = 0;
  }
  
  // Reset game state
  gameState.score = 0;
  gameState.tokensCollected = 0;
  gameState.checkpointsCompleted = 0;
  gameState.boostsAvailable = 0;
  gameState.gamePhase = "START";
  
  // Reinitialize entities
  initializeEntities();
  
  logGameInfo("START", { reset: true });
}

window.setControlMode = function(mode) {
  gameState.controlMode = mode;
  gameState.testStartTime = Date.now();
  
  // Update button states
  document.querySelectorAll('.control-button').forEach(btn => {
    btn.classList.remove('active');
  });
  
  if (mode === "HUMAN") {
    document.getElementById('humanModeBtn').classList.add('active');
  } else if (mode === "TEST_1") {
    document.getElementById('test_1_ModeBtn').classList.add('active');
  } else if (mode === "TEST_2") {
    document.getElementById('test_2_ModeBtn').classList.add('active');
  }
  
  logGameInfo("CONTROL_MODE_CHANGE", { mode });
};
