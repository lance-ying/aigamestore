import { gameState, TILE_SIZE, FRICTION, CANVAS_WIDTH, CANVAS_HEIGHT, DASH_SPEED, DASH_COOLDOWN, ATTACK_COOLDOWN, ATTACK_DURATION, COLORS } from './globals.js';
import { handleWorldCollision, checkEntityHit, checkRectOverlap } from './physics.js';
import { screenShake, dist } from './utils.js'; // Corrected: dist moved from physics.js to utils.js
import { createBloodEffect, createDustEffect, createSparkle } from './particles.js';

class Entity {
    constructor(x, y, width, height) {
        this.x = x;
        this.y = y;
        this.width = width;
        this.height = height;
        this.vx = 0;
        this.vy = 0;
        this.markedForDeletion = false;
        this.color = [255, 255, 255];
    }

    update(p) {
        // Basic physics update
        this.vx *= FRICTION;
        this.vy *= FRICTION;
    }

    render(p) {
        p.fill(this.color);
        p.rect(this.x, this.y, this.width, this.height);
    }
    
    getCenter() {
        return {
            x: this.x + this.width / 2,
            y: this.y + this.height / 2
        };
    }
}

export class Player extends Entity {
    constructor(x, y) {
        super(x, y, 20, 20);
        this.speed = 3;
        this.health = 100;
        this.maxHealth = 100;
        this.stamina = 100;
        this.maxStamina = 100;
        
        // State
        this.facing = 1; // 1 Right, -1 Left, 2 Down, -2 Up (Simplified: Angle)
        this.facingAngle = 0;
        this.isAttacking = false;
        this.attackTimer = 0;
        this.isDashing = false;
        this.dashTimer = 0;
        
        // I-Frames
        this.invincible = false;
        this.invincibleTimer = 0;
        
        // Visuals
        this.animTimer = 0;
    }

    update(p) {
        // Input Handling
        if (gameState.gamePhase === "PLAYING" && !this.isDashing) {
            this.handleInput(p);
        }

        // Dashing Logic
        if (this.isDashing) {
            this.dashTimer--;
            if (this.dashTimer <= 0) {
                this.isDashing = false;
                this.dashTimer = DASH_COOLDOWN;
            } else if (this.dashTimer > DASH_COOLDOWN - 10) {
                 // Active Dash phase (first 10 frames)
                 createDustEffect(this.x + this.width/2, this.y + this.height, this.color);
            } else {
                // Cooldown phase
                this.vx *= 0.5;
                this.vy *= 0.5;
            }
        } else {
            // Stamina regen
            if (this.dashTimer > 0) this.dashTimer--;
            this.stamina = Math.min(this.maxStamina, this.stamina + 0.5);
        }

        // Attacking Logic
        if (this.isAttacking) {
            this.attackTimer--;
            if (this.attackTimer <= 0) {
                this.isAttacking = false;
            }
            
            // Perform attack check once during the swing
            if (this.attackTimer === ATTACK_DURATION - 5) {
                this.checkAttackHit(p);
            }
        }

        // Physics
        handleWorldCollision(this);

        // Invincibility
        if (this.invincible) {
            this.invincibleTimer--;
            if (this.invincibleTimer <= 0) {
                this.invincible = false;
            }
        }
        
        // Death check
        if (this.health <= 0) {
            gameState.gamePhase = "GAME_OVER_LOSE";
            createBloodEffect(this.x, this.y, 20);
        }

        this.animTimer += 0.1;
    }

    handleInput(p) {
        let dx = 0;
        let dy = 0;
        const keyLeft = 37;
        const keyRight = 39;
        const keyUp = 38;
        const keyDown = 40;
        const keyShift = 16;
        const keyZ = 90;

        if (p.keyIsDown(keyLeft)) dx -= 1;
        if (p.keyIsDown(keyRight)) dx += 1;
        if (p.keyIsDown(keyUp)) dy -= 1;
        if (p.keyIsDown(keyDown)) dy += 1;

        // Normalize diagonal movement
        if (dx !== 0 || dy !== 0) {
            const length = Math.sqrt(dx*dx + dy*dy);
            dx /= length;
            dy /= length;
            
            this.vx += dx * 0.5;
            this.vy += dy * 0.5;
            
            // Limit speed
            const currentSpeed = Math.sqrt(this.vx*this.vx + this.vy*this.vy);
            if (currentSpeed > this.speed) {
                this.vx = (this.vx / currentSpeed) * this.speed;
                this.vy = (this.vy / currentSpeed) * this.speed;
            }
            
            this.facingAngle = Math.atan2(dy, dx);
            if (p.frameCount % 10 === 0) createDustEffect(this.x + 10, this.y + 20, [200, 200, 200]);
        }

        // Dash
        if (p.keyIsDown(keyShift) && this.dashTimer === 0 && this.stamina > 20) {
            this.startDash(dx, dy);
        }
        
        // Attack
        if (p.keyIsDown(keyZ) && !this.isAttacking) {
            this.startAttack();
        }
    }

    startDash(dx, dy) {
        if (dx === 0 && dy === 0) return; // Can't dash in place
        
        this.isDashing = true;
        this.dashTimer = DASH_COOLDOWN; // Set full duration + cooldown
        this.stamina -= 25;
        this.vx = dx * DASH_SPEED;
        this.vy = dy * DASH_SPEED;
        
        screenShake.add(0.2);
    }

    startAttack() {
        this.isAttacking = true;
        this.attackTimer = ATTACK_DURATION;
        screenShake.add(0.1);
    }

    checkAttackHit(p) {
        // Create a hitbox in front of player
        const reach = 30;
        const hitX = (this.x + 10) + Math.cos(this.facingAngle) * reach;
        const hitY = (this.y + 10) + Math.sin(this.facingAngle) * reach;
        const hitRadius = 25;

        // Check enemies
        gameState.enemies.forEach(enemy => {
            if (checkEntityHit(enemy, hitX, hitY, hitRadius)) {
                enemy.takeDamage(25, this.facingAngle);
                createBloodEffect(enemy.x + enemy.width/2, enemy.y + enemy.height/2);
                screenShake.add(0.2);
            }
        });
        
        // Check breakables (in interactables)
        gameState.interactables.forEach(obj => {
             if (obj.breakable && checkEntityHit(obj, hitX, hitY, hitRadius)) {
                 obj.break(p);
             }
        });
    }

    takeDamage(amount) {
        if (this.invincible || this.isDashing) return;
        
        this.health -= amount;
        this.invincible = true;
        this.invincibleTimer = 60; // 1 second
        screenShake.add(0.5);
        createBloodEffect(this.x + 10, this.y + 10, 10, [255, 0, 0]);
        
        // Log player info
        if (gameState.frameCount % 60 === 0) { // Throttle logs
             // Handled in main loop usually, but specific event logging is good
        }
    }

    render(p) {
        p.push();
        p.translate(this.x + 10, this.y + 10);
        
        // Shadow
        p.noStroke();
        p.fill(0, 0, 0, 100);
        p.ellipse(0, 8, 16, 6);

        // Flash white if invincible
        if (this.invincible && Math.floor(gameState.frameCount / 4) % 2 === 0) {
            p.fill(255);
        } else {
            p.fill(COLORS.ACCENT);
        }
        
        // Bobbing animation
        const bob = this.vx !== 0 || this.vy !== 0 ? Math.sin(this.animTimer) * 2 : 0;
        
        // Body (Armor)
        p.rectMode(p.CENTER);
        p.rect(0, 0 + bob, 14, 18, 2);
        
        // Helmet
        p.fill(200);
        p.rect(0, -8 + bob, 16, 14, 4);
        
        // Eyes (Visor)
        p.fill(0);
        p.rect(0, -8 + bob, 10, 3);
        
        // Sword Swing Animation
        if (this.isAttacking) {
            const swingProgress = 1 - (this.attackTimer / ATTACK_DURATION);
            const startAngle = this.facingAngle - p.PI / 3;
            const endAngle = this.facingAngle + p.PI / 3;
            const currentAngle = p.lerp(startAngle, endAngle, swingProgress);
            
            p.push();
            p.rotate(currentAngle);
            p.translate(15, 0);
            
            // Sword Blade
            p.fill(220);
            p.rect(0, 0, 25, 4);
            // Hilt
            p.fill(139, 69, 19);
            p.rect(-2, 0, 6, 8);
            
            // Swoosh effect
            p.noFill();
            p.stroke(255, 255, 255, 200 * (1-swingProgress));
            p.strokeWeight(2);
            p.arc(-15, 0, 50, 50, -p.PI/4, p.PI/4);
            
            p.pop();
        }

        p.pop();
        
        // Debug Hitbox
        // if (this.isAttacking) {
        //     p.noFill();
        //     p.stroke(255, 0, 0);
        //     const reach = 30;
        //     const hitX = (this.x + 10) + Math.cos(this.facingAngle) * reach;
        //     const hitY = (this.y + 10) + Math.sin(this.facingAngle) * reach;
        //     p.circle(hitX, hitY, 50);
        // }
    }
}

export class Enemy extends Entity {
    constructor(x, y, type) {
        super(x, y, 20, 20);
        this.type = type; // 'SKELETON', 'BAT', 'MAGE'
        this.health = 50;
        this.state = 'IDLE'; // IDLE, CHASE, ATTACK
        this.detectionRange = 200;
        this.attackRange = 30;
        this.knockback = {x: 0, y: 0};
        
        // Type specific stats
        if (type === 'BAT') {
            this.speed = 2.5;
            this.health = 30;
            this.color = [100, 80, 200];
            this.flying = true;
        } else if (type === 'MAGE') {
            this.speed = 1.0;
            this.health = 40;
            this.attackRange = 150;
            this.color = [180, 30, 30];
            this.cooldown = 0;
        } else { // SKELETON
            this.speed = 1.5;
            this.health = 60;
            this.color = [200, 200, 200];
        }
        
        this.animOffset = Math.random() * 100;
    }

    update(p) {
        if (!gameState.player) return;

        // Apply knockback friction
        this.x += this.knockback.x;
        this.y += this.knockback.y;
        this.knockback.x *= 0.8;
        this.knockback.y *= 0.8;
        
        // Behavior
        const d = dist(this.x, this.y, gameState.player.x, gameState.player.y);
        
        if (d < this.detectionRange) {
            this.state = 'CHASE';
        } else {
            this.state = 'IDLE';
        }

        if (this.state === 'CHASE') {
            // Move towards player
            if (d > this.attackRange) {
                const angle = Math.atan2(gameState.player.y - this.y, gameState.player.x - this.x);
                this.vx += Math.cos(angle) * 0.2;
                this.vy += Math.sin(angle) * 0.2;
                
                // Cap speed
                const speed = Math.sqrt(this.vx*this.vx + this.vy*this.vy);
                if (speed > this.speed) {
                    this.vx = (this.vx / speed) * this.speed;
                    this.vy = (this.vy / speed) * this.speed;
                }
            } else {
                // Attack logic
                this.performAttack(p);
            }
        }
        
        super.update(p);
        
        // Physics collision
        if (!this.flying) {
            handleWorldCollision(this);
        }
        
        // Resolve overlapping with other enemies (simple separation)
        gameState.enemies.forEach(other => {
            if (other !== this) {
                const distOther = dist(this.x, this.y, other.x, other.y);
                if (distOther < this.width) {
                    const angle = Math.atan2(this.y - other.y, this.x - other.x);
                    this.x += Math.cos(angle) * 1;
                    this.y += Math.sin(angle) * 1;
                }
            }
        });
    }

    performAttack(p) {
        if (this.type === 'MAGE') {
            if (this.cooldown <= 0) {
                // Shoot projectile
                const angle = Math.atan2(gameState.player.y - this.y, gameState.player.x - this.x);
                gameState.projectiles.push(new Projectile(this.x + 10, this.y + 10, angle, false));
                this.cooldown = 120; // 2 seconds
            } else {
                this.cooldown--;
            }
        } else {
            // Melee
            if (checkRectOverlap(this.x, this.y, this.width, this.height, gameState.player.x, gameState.player.y, gameState.player.width, gameState.player.height)) {
                gameState.player.takeDamage(10);
                // Knockback player slightly
                const angle = Math.atan2(gameState.player.y - this.y, gameState.player.x - this.x);
                gameState.player.vx += Math.cos(angle) * 10;
                gameState.player.vy += Math.sin(angle) * 10;
            }
        }
    }

    takeDamage(amount, angleFromAttacker) {
        this.health -= amount;
        this.knockback.x = Math.cos(angleFromAttacker) * 8;
        this.knockback.y = Math.sin(angleFromAttacker) * 8;
        
        // Flash red
        // Handled in render logic via visual effect usually, simple here
        
        if (this.health <= 0) {
            this.die(p);
        }
    }

    die(p) {
        this.markedForDeletion = true;
        gameState.score += (this.type === 'MAGE' ? 50 : 20);
        createBloodEffect(this.x + 10, this.y + 10, 15);
        
        // Chance to drop item
        if (Math.random() < 0.3) {
            gameState.pickups.push(new Pickup(this.x, this.y, Math.random() < 0.5 ? 'GOLD' : 'POTION'));
        }
    }

    render(p) {
        p.push();
        p.translate(this.x + 10, this.y + 10);
        
        // Bobbing
        const bob = Math.sin((gameState.frameCount + this.animOffset) * 0.1) * 3;
        
        // Shadow
        p.fill(0, 0, 0, 100);
        p.noStroke();
        p.ellipse(0, 10, 14, 6);
        
        if (this.type === 'BAT') {
            p.fill(this.color);
            p.translate(0, bob - 5); // Fly higher
            p.beginShape();
            p.vertex(0, 0);
            p.vertex(-10, -10);
            p.vertex(-5, 5);
            p.vertex(0, 0);
            p.vertex(5, 5);
            p.vertex(10, -10);
            p.endShape(p.CLOSE);
        } else if (this.type === 'SKELETON') {
            p.fill(220);
            p.translate(0, bob * 0.2);
            p.rectMode(p.CENTER);
            p.rect(0, 0, 14, 16);
            p.fill(0);
            p.rect(-3, -4, 3, 3);
            p.rect(3, -4, 3, 3);
        } else if (this.type === 'MAGE') {
            p.fill(this.color);
            p.translate(0, bob * 0.5);
            p.triangle(0, -10, -8, 10, 8, 10);
            p.fill(255, 200, 0); // Hat brim
            p.rectMode(p.CENTER);
            p.rect(0, -5, 18, 2);
        }
        
        p.pop();
    }
}

export class Projectile extends Entity {
    constructor(x, y, angle, isPlayerOwned) {
        super(x, y, 8, 8);
        this.speed = 6;
        this.vx = Math.cos(angle) * this.speed;
        this.vy = Math.sin(angle) * this.speed;
        this.isPlayerOwned = isPlayerOwned;
        this.life = 100;
    }

    update(p) {
        this.x += this.vx;
        this.y += this.vy;
        this.life--;
        
        if (this.life <= 0) this.markedForDeletion = true;
        
        // Wall collision
        const c = Math.floor((this.x + 4) / TILE_SIZE);
        const r = Math.floor((this.y + 4) / TILE_SIZE);
        const tile = gameState.world[r] && gameState.world[r][c];
        if (tile && tile.solid) {
            this.markedForDeletion = true;
            createSparkle(this.x, this.y);
            return;
        }

        // Entity collision
        if (this.isPlayerOwned) {
            // Player shooting logic (if implemented)
        } else {
            // Enemy shooting player
            if (checkRectOverlap(this.x, this.y, this.width, this.height, gameState.player.x, gameState.player.y, gameState.player.width, gameState.player.height)) {
                gameState.player.takeDamage(15);
                this.markedForDeletion = true;
            }
        }
    }

    render(p) {
        p.push();
        p.translate(this.x + 4, this.y + 4);
        p.rotate(Math.atan2(this.vy, this.vx));
        p.fill(this.isPlayerOwned ? [0, 255, 255] : [255, 100, 0]);
        p.noStroke();
        p.ellipse(0, 0, 10, 6);
        p.pop();
    }
}

export class Pickup extends Entity {
    constructor(x, y, type) {
        super(x, y, 16, 16);
        this.type = type; // 'GOLD', 'POTION', 'KEY'
        this.bobOffset = Math.random() * 100;
    }

    update(p) {
        // Magnetic effect if player is close
        if (dist(this.x, this.y, gameState.player.x, gameState.player.y) < 60) {
            this.x += (gameState.player.x - this.x) * 0.05;
            this.y += (gameState.player.y - this.y) * 0.05;
        }

        if (checkRectOverlap(this.x, this.y, this.width, this.height, gameState.player.x, gameState.player.y, gameState.player.width, gameState.player.height)) {
            this.collect();
        }
    }

    collect() {
        this.markedForDeletion = true;
        createSparkle(this.x + 8, this.y + 8);
        
        if (this.type === 'GOLD') {
            gameState.score += 10;
        } else if (this.type === 'POTION') {
            gameState.player.health = Math.min(gameState.player.maxHealth, gameState.player.health + 25);
        } else if (this.type === 'KEY') {
            gameState.keys++;
            gameState.score += 100;
        }
    }

    render(p) {
        p.push();
        p.translate(this.x + 8, this.y + 8 + Math.sin((gameState.frameCount + this.bobOffset) * 0.1) * 3);
        
        if (this.type === 'GOLD') {
            p.fill(255, 215, 0);
            p.circle(0, 0, 10);
            p.fill(255, 255, 0);
            p.circle(-2, -2, 3);
        } else if (this.type === 'POTION') {
            p.fill(255, 0, 0);
            p.rectMode(p.CENTER);
            p.rect(0, 2, 8, 10, 2);
            p.fill(200);
            p.rect(0, -4, 4, 3);
        } else if (this.type === 'KEY') {
            p.fill(192, 192, 192);
            p.rotate(p.PI / 4);
            p.rectMode(p.CENTER);
            p.rect(0, 0, 4, 12);
            p.circle(0, -6, 8);
        }
        
        p.pop();
    }
}

export class Interactable extends Entity {
    constructor(x, y, type) {
        super(x, y, TILE_SIZE, TILE_SIZE);
        this.type = type; // 'CHEST', 'DOOR', 'EXIT'
        this.opened = false;
        this.locked = (type === 'DOOR');
    }

    update(p) {
        if (this.opened) return;

        // Check interaction
        const d = dist(this.x + this.width/2, this.y + this.height/2, gameState.player.x + 10, gameState.player.y + 10);
        if (d < 40 && p.keyIsDown(32)) { // SPACE
            this.interact();
        }
        
        // Auto interact with exit
        if (this.type === 'EXIT' && d < 20) {
            gameState.gamePhase = "GAME_OVER_WIN";
        }
    }

    interact() {
        if (this.opened) return;
        
        if (this.type === 'DOOR') {
            if (gameState.keys > 0) {
                gameState.keys--;
                this.opened = true;
                this.width = 0; // Remove collision
                this.height = 0;
                createSparkle(this.x + TILE_SIZE/2, this.y + TILE_SIZE/2);
            }
        } else if (this.type === 'CHEST') {
            this.opened = true;
            createSparkle(this.x + TILE_SIZE/2, this.y + TILE_SIZE/2);
            gameState.score += 500;
            // Spawn loot
            for(let i=0; i<5; i++) {
                const loot = new Pickup(this.x + TILE_SIZE/2, this.y + TILE_SIZE/2, 'GOLD');
                loot.vx = Math.random() * 4 - 2;
                loot.vy = Math.random() * 4 - 2;
                gameState.pickups.push(loot);
            }
        }
    }

    render(p) {
        if (this.opened && this.type !== 'EXIT') {
            // Render opened state
            p.fill(100);
            p.rect(this.x, this.y, this.width || TILE_SIZE, this.height || TILE_SIZE);
            return;
        }

        if (this.type === 'DOOR') {
            p.fill(139, 69, 19);
            p.rect(this.x, this.y, TILE_SIZE, TILE_SIZE);
            p.fill(0);
            p.rect(this.x + 5, this.y + 5, TILE_SIZE - 10, TILE_SIZE - 10);
            // Lock
            p.fill(255, 215, 0);
            p.circle(this.x + TILE_SIZE/2, this.y + TILE_SIZE/2, 8);
        } else if (this.type === 'CHEST') {
            p.fill(160, 82, 45);
            p.rect(this.x + 5, this.y + 10, TILE_SIZE - 10, TILE_SIZE - 20);
            p.fill(255, 215, 0);
            p.rect(this.x + TILE_SIZE/2 - 2, this.y + 10, 4, 10);
        } else if (this.type === 'EXIT') {
            p.fill(0);
            p.rect(this.x, this.y, TILE_SIZE, TILE_SIZE);
            // Stairs pattern
            p.stroke(50);
            for(let i=0; i<4; i++) {
                p.line(this.x, this.y + i*10, this.x + TILE_SIZE, this.y + i*10);
            }
            p.noStroke();
        }
    }
}
