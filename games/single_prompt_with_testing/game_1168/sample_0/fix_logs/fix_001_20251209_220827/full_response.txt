import { gameState, CANVAS_WIDTH, CANVAS_HEIGHT, TILE_SIZE } from './globals.js';
import { Platform, Enemy, Collectible } from './entities.js';

export function generateLevel(p) {
    // Reset
    gameState.platforms = [];
    gameState.entities = [];
    gameState.smokeY = CANVAS_HEIGHT + 200;
    
    // Tower Parameters
    const towerWidth = CANVAS_WIDTH; // full width
    const towerHeight = gameState.worldHeight;
    const floorHeight = 100; // Vertical distance between "floors" roughly
    
    // 1. Create Floor and Walls
    // Bottom floor
    gameState.platforms.push(new Platform(0, CANVAS_HEIGHT - 40, CANVAS_WIDTH, 40));
    
    // Walls (left and right)
    // We create tall walls in segments
    for (let y = -towerHeight; y < CANVAS_HEIGHT; y += 400) {
        gameState.platforms.push(new Platform(0, y, 40, 400)); // Left Wall
        gameState.platforms.push(new Platform(CANVAS_WIDTH - 40, y, 40, 400)); // Right Wall
    }
    
    // 2. Generate Platforms
    // We go from bottom to top
    let currentY = CANVAS_HEIGHT - 150;
    
    while (currentY > -towerHeight + 200) {
        // Generate a row of platforms
        // Pattern: Randomly scatter 1-3 platforms at this Y level
        let numPlats = Math.floor(p.random(1, 4));
        
        // Ensure playable gaps
        let xPositions = [];
        let gap = (CANVAS_WIDTH - 80) / numPlats;
        
        for (let i = 0; i < numPlats; i++) {
            let pWidth = p.random(60, 150);
            // Center within its segment
            let segmentX = 40 + (i * gap);
            let pX = segmentX + p.random(0, gap - pWidth);
            
            gameState.platforms.push(new Platform(pX, currentY, pWidth, 20));
            
            // 3. Spawn Entities on Platform
            if (p.random() < 0.4) {
                // Spawn Enemy
                let type = p.random() > 0.5 ? "SLIME" : (p.random() > 0.5 ? "BAT" : "TURRET");
                new Enemy(pX + pWidth/2, currentY - 40, type);
            } else if (p.random() < 0.1) {
                // Spawn Item
                new Collectible(pX + pWidth/2, currentY - 30, "HEALTH");
            }
        }
        
        // Move up
        currentY -= p.random(90, 130);
    }
    
    // 4. Boss Arena at Top
    gameState.platforms.push(new Platform(40, -towerHeight + 100, CANVAS_WIDTH - 80, 20));
    // Boss entity could be added here
    new Enemy(CANVAS_WIDTH/2, -towerHeight + 50, "TURRET"); // Placeholder boss
    
    // 5. Create Player
    // The Player is instantiated in game.js after level generation.
    // This file does not need to import or instantiate the Player class.
}
