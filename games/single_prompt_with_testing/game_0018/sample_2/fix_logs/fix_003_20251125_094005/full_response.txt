<analysis>
- Issue identified: The game crashes when track/tunnel segments are depleted. The `updateWorld` function in `world.js` tries to access `trackSegments[length-1]` when the array might be empty (after all segments scroll past and get removed). It also uses the stored `z` property instead of the actual mesh position, causing improper placement of new segments.
- Files affected: world.js
- Changes needed: 
  1. Check if segment arrays are empty before accessing last element
  2. Use actual mesh position (`lastSegment.mesh.position.z`) instead of stored `z` value
  3. Change `if` to `while` loops as the comments indicate to handle multiple segment spawns in one frame
</analysis>

import { gameState } from './globals.js';
import { TrackSegment, TunnelSegment } from './entities.js';

export function initializeWorld() {
  // Create initial track segments
  for (let i = 0; i < 10; i++) {
    const segment = new TrackSegment(-i * 10);
    gameState.trackSegments.push(segment);
  }
  
  // Create initial tunnel segments
  for (let i = 0; i < 10; i++) {
    const segment = new TunnelSegment(-i * 10);
    gameState.tunnelSegments.push(segment);
  }
}

export function updateWorld(deltaTime) {
  // Spawn new track segments as needed - use while loop to handle multiple segments
  while (gameState.trackSegments.length < 10) {
    let newZ;
    if (gameState.trackSegments.length === 0) {
      // If array is empty, start from a default position
      newZ = -10;
    } else {
      const lastSegment = gameState.trackSegments[gameState.trackSegments.length - 1];
      // Use the current mesh position instead of the stored z value
      newZ = lastSegment.mesh.position.z - 10;
    }
    const segment = new TrackSegment(newZ);
    gameState.trackSegments.push(segment);
  }
  
  // Spawn new tunnel segments as needed - use while loop to handle multiple segments
  while (gameState.tunnelSegments.length < 10) {
    let newZ;
    if (gameState.tunnelSegments.length === 0) {
      // If array is empty, start from a default position
      newZ = -10;
    } else {
      const lastSegment = gameState.tunnelSegments[gameState.tunnelSegments.length - 1];
      // Use the current mesh position instead of the stored z value
      newZ = lastSegment.mesh.position.z - 10;
    }
    const segment = new TunnelSegment(newZ);
    gameState.tunnelSegments.push(segment);
  }
}
