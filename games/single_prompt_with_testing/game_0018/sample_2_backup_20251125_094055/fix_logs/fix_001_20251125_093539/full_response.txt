<analysis>
- Issue identified: 
  1. **Game freeze**: The `updateWorld()` function only spawns one track/tunnel segment per frame even if multiple are removed. With an if-statement instead of while-loop, track segments can run out faster than they're replaced, causing the world to disappear and potentially freeze the game.
  2. **UI not in canvas**: The UI overlay is positioned absolutely at `top: 0`, which places it at the top of the page rather than aligned with the canvas. The canvas is centered in the page using flexbox, so the UI needs to be positioned relative to the canvas location or wrapped in the same container.

- Files affected: 
  - `ui.js` - needs to position UI overlay relative to canvas
  - `world.js` - needs to use while-loop instead of if-statement for spawning segments
  - `index.html` - needs to wrap canvas and UI together

- Changes needed:
  - Change `updateWorld()` to use while-loops instead of if-statements so multiple segments spawn when needed
  - Position UI overlay to be relative to the canvas by placing it after the renderer canvas in DOM and adjusting positioning
</analysis>

import { gameState } from './globals.js';
import { TrackSegment, TunnelSegment } from './entities.js';

export function initializeWorld() {
  // Create initial track segments
  for (let i = 0; i < 10; i++) {
    const segment = new TrackSegment(-i * 10);
    gameState.trackSegments.push(segment);
  }
  
  // Create initial tunnel segments
  for (let i = 0; i < 10; i++) {
    const segment = new TunnelSegment(-i * 10);
    gameState.tunnelSegments.push(segment);
  }
}

export function updateWorld(deltaTime) {
  // Spawn new track segments as needed - use while loop to handle multiple segments
  while (gameState.trackSegments.length < 10) {
    const lastSegment = gameState.trackSegments[gameState.trackSegments.length - 1];
    const newZ = lastSegment.z - 10;
    const segment = new TrackSegment(newZ);
    gameState.trackSegments.push(segment);
  }
  
  // Spawn new tunnel segments as needed - use while loop to handle multiple segments
  while (gameState.tunnelSegments.length < 10) {
    const lastSegment = gameState.tunnelSegments[gameState.tunnelSegments.length - 1];
    const newZ = lastSegment.z - 10;
    const segment = new TunnelSegment(newZ);
    gameState.tunnelSegments.push(segment);
  }
}

import { gameState, WIN_SCORE } from './globals.js';

let uiOverlay = null;

export function setupUI() {
  uiOverlay = document.createElement('div');
  uiOverlay.style.position = 'absolute';
  uiOverlay.style.top = '0';
  uiOverlay.style.left = '0';
  uiOverlay.style.width = '600px';
  uiOverlay.style.height = '400px';
  uiOverlay.style.pointerEvents = 'none';
  uiOverlay.style.fontFamily = 'Arial, sans-serif';
  
  // Find the canvas and insert UI overlay right after it
  const canvas = gameState.renderer.domElement;
  canvas.style.position = 'relative';
  
  // Create a wrapper for canvas and UI
  const wrapper = document.createElement('div');
  wrapper.style.position = 'relative';
  wrapper.style.width = '600px';
  wrapper.style.height = '400px';
  wrapper.style.margin = '0 auto';
  
  // Move canvas into wrapper
  canvas.parentNode.insertBefore(wrapper, canvas);
  wrapper.appendChild(canvas);
  wrapper.appendChild(uiOverlay);
}
