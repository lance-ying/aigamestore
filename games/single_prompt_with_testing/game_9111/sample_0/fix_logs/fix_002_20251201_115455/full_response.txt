<analysis>
- Issue identified: When the egg stack hits the front of an obstacle, the colliding eggs are removed correctly, but the player's vertical position relies on the `currentGroundY` which only updates when the player's *body* (left edge) passes the obstacle's left edge. This gap causes the player to fall momentarily into the obstacle's space, leading to subsequent eggs colliding with the same obstacle or the player dying, even if they had enough height to clear it.
- Files affected: `entities.js`, `physics.js`
- Changes needed:
  1.  In `entities.js` (Player class): Add a `groundedObstacle` property to track an obstacle the player has "latched" onto. Update the `update()` loop to use this property to snap `currentGroundY` to the obstacle's top even if only overlapping horizontally (not yet fully mounted).
  2.  In `physics.js`: When eggs collide with an obstacle and break, immediately set `player.groundedObstacle` to that obstacle. This ensures that in the next frame, the player is considered "supported" by that obstacle, preventing them from falling into it.
  3.  (Minor fix) In `physics.js`: Fixed projectiles not actually destroying obstacles (missing `markedForDeletion`).
</analysis>

import { GAME_OPTS, COLORS, CANVAS_HEIGHT, gameState, CANVAS_WIDTH } from './globals.js';
import { generateID } from './utils.js';

export class Entity {
    constructor(x, y, w, h) {
        this.id = generateID();
        this.x = x;
        this.y = y;
        this.w = w;
        this.h = h;
        this.markedForDeletion = false;
    }

    update(dt) {}
    render(p) {}
}

export class Player extends Entity {
    constructor(x, y) {
        super(x, y, GAME_OPTS.playerSize, GAME_OPTS.playerSize);
        this.eggs = 0; // Current stack height (number of eggs)
        this.vy = 0;
        this.isGrounded = true;
        this.groundY = CANVAS_HEIGHT - GAME_OPTS.groundHeight;
        this.currentGroundY = this.groundY; // Dynamic ground level based on obstacles
        this.groundedObstacle = null; // Obstacle we are currently climbing/glued to
        this.feverShotTimer = 0;
        
        // Animation
        this.rotation = 0;
        this.bounce = 0;
        
        gameState.player = this;
    }

    update(dt) {
        // Fever mode handling
        if (gameState.feverMode) {
            gameState.feverTimer--;
            if (gameState.feverTimer <= 0) {
                this.deactivateFever();
            }
            
            // Auto-shoot in fever mode
            this.feverShotTimer++;
            if (this.feverShotTimer > 10) {
                this.shoot();
                this.feverShotTimer = 0;
            }
        }

        // Determine current ground level (handle standing on obstacles)
        this.currentGroundY = this.groundY;
        
        // Check sticky ground first (from physics collision)
        if (this.groundedObstacle) {
            // If obstacle is gone or we moved past it/away from it, release
            if (this.groundedObstacle.markedForDeletion || 
                !(this.x + this.w > this.groundedObstacle.x && this.x < this.groundedObstacle.x + this.groundedObstacle.w)) {
                this.groundedObstacle = null;
            } else {
                this.currentGroundY = this.groundedObstacle.y;
            }
        }
        
        for (const obs of gameState.obstacles) {
            // Check if we are "on" the obstacle.
            // We consider the player "on" the obstacle if their left edge has passed the obstacle's left edge.
            // This ensures we take collision damage from the front face before mounting.
            if (this.x >= obs.x && this.x < obs.x + obs.w) {
                // We are over this obstacle.
                // If this obstacle is higher than our current ground (meaning smaller Y), snap to it.
                if (obs.y < this.currentGroundY) {
                    this.currentGroundY = obs.y;
                }
            }
        }

        // Calculate target Y based on egg count and current ground
        const targetY = this.currentGroundY - (this.eggs * GAME_OPTS.eggSize) - this.h;
        
        // Smooth movement to target Y (gravity-like but snappy)
        this.y = targetY; 
        
        // Bounce animation
        if (this.bounce > 0) this.bounce -= 0.5;
    }

    layEgg() {
        this.eggs++;
        this.bounce = 5;
    }

    removeEgg(count = 1) {
        this.eggs = Math.max(0, this.eggs - count);
    }

    resetEggs() {
        this.eggs = 0;
    }

    activateFever() {
        gameState.feverMode = true;
        gameState.feverTimer = GAME_OPTS.feverDuration;
        gameState.landings = 0; // Reset counter
    }

    deactivateFever() {
        gameState.feverMode = false;
    }

    shoot() {
        const p = new Projectile(this.x + this.w, this.y + this.h/2);
        gameState.projectiles.push(p);
    }
    
    getRect() {
        return {x: this.x, y: this.y, w: this.w, h: this.h};
    }
    
    // Helper to get rects of all eggs for collision
    getEggRects() {
        let rects = [];
        for (let i = 0; i < this.eggs; i++) {
            rects.push({
                x: this.x,
                // Egg 0 is at bottom, Egg N-1 is just below player
                // Use currentGroundY so eggs sit on the obstacle if we are on it
                y: this.currentGroundY - ((i + 1) * GAME_OPTS.eggSize),
                w: GAME_OPTS.eggSize,
                h: GAME_OPTS.eggSize,
                index: i
            });
        }
        return rects;
    }

    render(p) {
        p.push();
        
        // Draw Eggs
        for (let i = 0; i < this.eggs; i++) {
            // Use currentGroundY for rendering too
            const eggY = this.currentGroundY - ((i + 1) * GAME_OPTS.eggSize);
            p.fill(COLORS.egg);
            p.stroke(COLORS.eggBorder);
            p.strokeWeight(2);
            p.rect(this.x, eggY, GAME_OPTS.eggSize, GAME_OPTS.eggSize, 5);
            
            // Egg detail
            p.noStroke();
            p.fill(240);
            p.circle(this.x + GAME_OPTS.eggSize/2, eggY + GAME_OPTS.eggSize/3, 5);
        }

        // Draw Player
        p.translate(this.x + this.w/2, this.y + this.h/2 - this.bounce);
        
        // Fever rotation/shake
        if (gameState.feverMode) {
            p.rotate(Math.sin(p.frameCount * 0.5) * 0.1);
            p.scale(1.1);
        }
        
        p.fill(COLORS.player);
        p.stroke(0);
        p.strokeWeight(2);
        p.rectMode(p.CENTER);
        p.rect(0, 0, this.w, this.h, 8); // Rounded bird

        // Face
        p.fill(COLORS.playerFace);
        p.noStroke();
        // Eye
        p.circle(5, -5, 5);
        // Beak
        p.fill(255, 100, 0);
        p.triangle(10, 0, 20, 2, 10, 5);
        // Wing
        p.stroke(0);
        p.strokeWeight(1);
        p.noFill();
        p.arc(-5, 5, 15, 10, 0, p.PI);

        p.pop();
    }
}

export class Obstacle extends Entity {
    constructor(x, w, h, type = "WALL") {
        // Obstacles are anchored to the ground
        const groundY = CANVAS_HEIGHT - GAME_OPTS.groundHeight;
        super(x, groundY - h, w, h);
        this.type = type; // WALL, GAP (though gap is just lack of obstacle)
        this.active = true;
    }

    update(speed) {
        this.x -= speed;
        if (this.x + this.w < 0) {
            this.markedForDeletion = true;
        }
    }

    render(p) {
        if (!this.active) return;
        
        p.push();
        p.fill(COLORS.obstacle);
        p.stroke(COLORS.obstacleBorder);
        p.strokeWeight(2);
        
        // Draw brick pattern
        p.rect(this.x, this.y, this.w, this.h);
        
        p.stroke(60);
        p.strokeWeight(1);
        const brickH = 20;
        for (let by = this.y; by < this.y + this.h; by += brickH) {
            p.line(this.x, by, this.x + this.w, by);
            // Staggered vertical lines would be nice but simple lines are ok
        }
        
        p.pop();
    }
}

export class Projectile extends Entity {
    constructor(x, y) {
        super(x, y, 15, 10);
        this.speed = 15;
    }
    
    update() {
        this.x += this.speed;
        if (this.x > CANVAS_WIDTH) this.markedForDeletion = true;
    }
    
    render(p) {
        p.push();
        p.fill(255, 50, 50);
        p.noStroke();
        p.ellipse(this.x, this.y, this.w, this.h);
        p.fill(255, 200, 50, 150);
        p.ellipse(this.x - 5, this.y, this.w, this.h); // Trail
        p.pop();
    }
}

export class Particle extends Entity {
    constructor(x, y, color) {
        super(x, y, randomRange(2, 6), randomRange(2, 6));
        this.vx = randomRange(-3, 3);
        this.vy = randomRange(-5, -1);
        this.gravity = 0.4;
        this.color = color;
        this.life = 60;
    }
    
    update() {
        this.x += this.vx;
        this.y += this.vy;
        this.vy += this.gravity;
        this.life--;
        if (this.life <= 0) this.markedForDeletion = true;
    }
    
    render(p) {
        p.push();
        p.noStroke();
        const alpha = p.map(this.life, 0, 60, 0, 255);
        const c = p.color(this.color[0], this.color[1], this.color[2]);
        c.setAlpha(alpha);
        p.fill(c);
        p.rect(this.x, this.y, this.w, this.w);
        p.pop();
    }
}

import { gameState, GAME_OPTS, COLORS } from './globals.js';
import { checkRectOverlap, pointInRect } from './utils.js';
import { Particle } from './entities.js';

export function updatePhysics(p) {
    const player = gameState.player;
    if (!player) return;

    // Move everything
    let currentSpeed = gameState.feverMode ? GAME_OPTS.feverSpeed : GAME_OPTS.scrollSpeed;
    
    // Update distance
    gameState.distance += currentSpeed;
    gameState.score = Math.floor(gameState.distance / 10);
    
    // Check Win
    if (gameState.distance >= GAME_OPTS.levelLength) {
        gameState.gamePhase = "GAME_OVER_WIN";
        return;
    }

    // Update Obstacles
    for (let i = gameState.obstacles.length - 1; i >= 0; i--) {
        const obs = gameState.obstacles[i];
        obs.update(currentSpeed);
        
        if (obs.markedForDeletion) {
            gameState.obstacles.splice(i, 1);
            continue;
        }

        // Collision Detection
        
        // 1. Check Projectiles vs Obstacle
        for (let j = gameState.projectiles.length - 1; j >= 0; j--) {
            const proj = gameState.projectiles[j];
            if (checkRectOverlap(proj, obs)) {
                // Destroy obstacle
                destroyObstacle(obs, p);
                // Destroy projectile
                proj.markedForDeletion = true;
                obs.markedForDeletion = true; // Mark obstacle for deletion
                break; // Obstacle gone
            }
        }
        if (obs.markedForDeletion) {
            gameState.obstacles.splice(i, 1);
            continue;
        }

        // 2. Check Player Fever Collision (Smashes obstacles)
        if (gameState.feverMode) {
            // Check player body and entire egg stack
            const playerRect = player.getRect();
            // Combine player + stack rect for simplicity or check individually
            // Actually in fever mode, player destroys everything he touches
            
            let collision = checkRectOverlap(playerRect, obs);
            if (!collision) {
                // Check eggs
                const eggRects = player.getEggRects();
                for (let egg of eggRects) {
                    if (checkRectOverlap(egg, obs)) {
                        collision = true;
                        break;
                    }
                }
            }
            
            if (collision) {
                destroyObstacle(obs, p);
                gameState.obstacles.splice(i, 1);
                continue;
            }
        } else {
            // Normal Mode Collision
            
            // A. Check Bird Body
            const birdRect = player.getRect();
            // Shrink hit box slightly for forgiveness
            const hitBox = {
                x: birdRect.x + 5,
                y: birdRect.y + 5,
                w: birdRect.w - 10,
                h: birdRect.h - 10
            };
            
            if (checkRectOverlap(hitBox, obs)) {
                // Bird hit a wall -> Game Over
                gameState.gamePhase = "GAME_OVER_LOSE";
                createExplosion(player.x, player.y, COLORS.player, p);
                return;
            }
            
            // B. Check Eggs (from bottom up)
            // If an egg hits, it breaks. If multiple hit, they all break.
            const eggRects = player.getEggRects();
            let eggsBroken = 0;
            
            for (let e = 0; e < eggRects.length; e++) {
                const egg = eggRects[e];
                // Check egg overlap
                if (checkRectOverlap(egg, obs)) {
                    // This egg hit the wall. 
                    // Physics logic: In this game, if the stack hits a wall, the eggs that hit are stripped away.
                    // The bird falls to the new stack height.
                    eggsBroken++;
                    createExplosion(egg.x + egg.w, egg.y + egg.h/2, COLORS.egg, p);
                }
            }
            
            if (eggsBroken > 0) {
                // Remove the bottom-most eggs involved?
                // Actually, if egg at index 0 (bottom) hits, it breaks.
                // If egg at index 1 hits, it breaks.
                // So we reduce egg count by eggsBroken.
                player.removeEgg(eggsBroken);
                // Reset combo/landings if we hit a wall? Usually yes.
                gameState.landings = 0;
                
                // IMPORTANT: We hit the wall and paid the egg toll.
                // Snap to this obstacle immediately so we don't fall into it next frame.
                player.groundedObstacle = obs;
            }
        }
    }
    
    // Update Projectiles
    for (let i = gameState.projectiles.length - 1; i >= 0; i--) {
        const proj = gameState.projectiles[i];
        proj.update();
        if (proj.markedForDeletion) {
            gameState.projectiles.splice(i, 1);
        }
    }
    
    // Update Particles
    for (let i = gameState.particles.length - 1; i >= 0; i--) {
        const part = gameState.particles[i];
        part.update();
        if (part.markedForDeletion) {
            gameState.particles.splice(i, 1);
        }
    }
    
    // Landing Logic (for Fever)
    // Check if player is on ground (eggs == 0) and wasn't previously
    // We need to track previous state. 
    // Simplified: Check if eggs == 0. If yes, and we just came from eggs > 0 state?
    // We can track `lastFrameEggs`.
    
    if (player.eggs === 0 && gameState.lastFrameEggs > 0) {
        // Landed
        gameState.landings++;
        // Visual cue
        createExplosion(player.x + player.w/2, player.groundY, COLORS.ground, p, 5);
        
        if (gameState.landings >= GAME_OPTS.landingTarget) {
            player.activateFever();
        }
    } else if (player.eggs > 0 && gameState.lastFrameEggs === 0) {
        // Took off
    }
    
    gameState.lastFrameEggs = player.eggs;
}

function destroyObstacle(obs, p) {
    createExplosion(obs.x + obs.w/2, obs.y + obs.h/2, COLORS.obstacle, p, 10);
}

function createExplosion(x, y, color, p, count = 10) {
    for (let i = 0; i < count; i++) {
        gameState.particles.push(new Particle(x, y, color));
    }
}
