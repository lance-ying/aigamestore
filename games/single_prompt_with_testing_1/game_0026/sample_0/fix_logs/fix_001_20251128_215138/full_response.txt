<analysis>
- Issue identified: The game crashes when pressing ENTER because the `startGame()` function uses CommonJS `require()` calls which don't work in a browser ES6 module environment. The code is mixing ES6 imports at the top of the file with `require()` calls inside functions.
- Files affected: input.js
- Changes needed: Add missing imports at the top of input.js and remove all `require()` calls from inside the `startGame()` and `prepareNextEncounter()` functions.

// input.js - Input handling
import { gameState, GAME_PHASES, HERO_CLASSES } from './globals.js';
import { drawCards, getStarterDeck, createCardDatabase, shuffleDeck } from './cards.js';
import { executePlayerTurn, startNewEncounter } from './combat.js';
import { Hero } from './entities.js';

export function handleKeyPressed(p, key, keyCode) {
  // Log input
  p.logs.inputs.push({
    input_type: "keyPressed",
    data: { key, keyCode },
    framecount: p.frameCount,
    timestamp: Date.now()
  });
  
  // Game phase transitions
  if (keyCode === 13 && gameState.gamePhase === GAME_PHASES.START) { // ENTER
    startGame(p);
    return;
  }
  
  if (keyCode === 82) { // R - Restart
    restartGame(p);
    return;
  }
  
  if (keyCode === 27 && gameState.gamePhase === GAME_PHASES.PLAYING) { // ESC - Pause
    gameState.gamePhase = GAME_PHASES.PAUSED;
    p.logs.game_info.push({
      data: { gamePhase: gameState.gamePhase },
      framecount: p.frameCount,
      timestamp: Date.now()
    });
    return;
  }
  
  if (keyCode === 27 && gameState.gamePhase === GAME_PHASES.PAUSED) { // ESC - Unpause
    gameState.gamePhase = GAME_PHASES.PLAYING;
    p.logs.game_info.push({
      data: { gamePhase: gameState.gamePhase },
      framecount: p.frameCount,
      timestamp: Date.now()
    });
    return;
  }
  
  // Gameplay input
  if (gameState.gamePhase === GAME_PHASES.PLAYING) {
    handleGameplayInput(p, keyCode);
  }
}

function handleGameplayInput(p, keyCode) {
  // Shop navigation
  if (gameState.shopOpen) {
    if (keyCode === 37) { // Left arrow
      gameState.selectedShopIndex = Math.max(0, gameState.selectedShopIndex - 1);
    } else if (keyCode === 39) { // Right arrow
      gameState.selectedShopIndex = Math.min(gameState.availableCards.length - 1, gameState.selectedShopIndex + 1);
    } else if (keyCode === 32) { // Space - Buy
      purchaseCard(p);
    } else if (keyCode === 90) { // Z - Close shop
      gameState.shopOpen = false;
      prepareNextEncounter(p);
    }
    return;
  }
  
  // Card selection
  if (gameState.turnPhase === "SELECT_CARDS") {
    if (keyCode === 37) { // Left arrow
      gameState.selectedHandIndex = Math.max(0, gameState.selectedHandIndex - 1);
    } else if (keyCode === 39) { // Right arrow
      gameState.selectedHandIndex = Math.min(gameState.hand.length - 1, gameState.selectedHandIndex + 1);
    } else if (keyCode === 32) { // Space
      if (gameState.selectedCards.length === 3) {
        // Execute turn
        executePlayerTurn(p);
      } else if (gameState.selectedHandIndex >= 0 && gameState.selectedHandIndex < gameState.hand.length) {
        // Select card
        const card = gameState.hand[gameState.selectedHandIndex];
        if (!gameState.selectedCards.includes(card) && gameState.selectedCards.length < 3) {
          gameState.selectedCards.push(card);
        }
      }
    } else if (keyCode === 90) { // Z - Cancel
      if (gameState.selectedCards.length > 0) {
        gameState.selectedCards.pop();
      }
    } else if (keyCode === 16) { // Shift - Details
      gameState.detailCardIndex = gameState.selectedHandIndex;
    }
  }
}

export function handleKeyReleased(p, key, keyCode) {
  if (keyCode === 16) { // Shift released
    gameState.detailCardIndex = -1;
  }
}

function startGame(p) {
  gameState.gamePhase = GAME_PHASES.PLAYING;
  
  // Initialize heroes
  gameState.heroes = [
    new Hero("Armilly", HERO_CLASSES.WARRIOR, 100, 150),
    new Hero("Copernica", HERO_CLASSES.MAGE, 100, 240),
    new Hero("Galleo", HERO_CLASSES.ROGUE, 100, 330)
  ];
  
  gameState.player = gameState.heroes[0]; // For testing compatibility
  
  // Initialize deck
  gameState.deck = getStarterDeck();
  shuffleDeck(gameState.deck);
  gameState.hand = [];
  gameState.selectedCards = [];
  gameState.discardPile = [];
  
  // Initialize shop cards
  const allCards = createCardDatabase();
  gameState.availableCards = allCards.slice(0, 10).map(c => Object.assign({}, c));
  
  // Reset state
  gameState.currentEncounter = 0;
  gameState.gold = 100;
  gameState.battlesWon = 0;
  gameState.level = 1;
  gameState.experience = 0;
  gameState.shopOpen = false;
  
  // Start first encounter
  startNewEncounter(p);
  drawCards(gameState.deck, gameState.hand, gameState.discardPile, 5);
  
  p.logs.game_info.push({
    data: { gamePhase: gameState.gamePhase, message: "Game started" },
    framecount: p.frameCount,
    timestamp: Date.now()
  });
}

function restartGame(p) {
  gameState.gamePhase = GAME_PHASES.START;
  gameState.heroes = [];
  gameState.enemies = [];
  gameState.deck = [];
  gameState.hand = [];
  gameState.selectedCards = [];
  gameState.discardPile = [];
  gameState.availableCards = [];
  
  p.logs.game_info.push({
    data: { gamePhase: gameState.gamePhase, message: "Game restarted" },
    framecount: p.frameCount,
    timestamp: Date.now()
  });
}

function purchaseCard(p) {
  if (gameState.selectedShopIndex >= 0 && gameState.selectedShopIndex < gameState.availableCards.length) {
    const card = gameState.availableCards[gameState.selectedShopIndex];
    if (gameState.gold >= card.cost) {
      gameState.gold -= card.cost;
      const newCard = Object.assign({}, card);
      gameState.deck.push(newCard);
      gameState.availableCards.splice(gameState.selectedShopIndex, 1);
      gameState.selectedShopIndex = Math.min(gameState.selectedShopIndex, gameState.availableCards.length - 1);
    }
  }
}

function prepareNextEncounter(p) {
  // Heal heroes partially
  for (const hero of gameState.heroes) {
    if (hero.isAlive()) {
      hero.heal(30);
    }
  }
  
  // Reset deck
  gameState.deck.push(...gameState.discardPile);
  gameState.discardPile = [];
  shuffleDeck(gameState.deck);
  gameState.hand = [];
  
  // Start next encounter
  startNewEncounter(p);
  drawCards(gameState.deck, gameState.hand, gameState.discardPile, 5);
}
