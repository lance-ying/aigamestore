<analysis>
- Issue identified: Level 4 is unbeatable because the procedural generation logic for levels greater than 3 does not coordinate wall placement with path turns, resulting in walls blocking the path at corners. Additionally, the user specifically requested holes in the walls for Level 4 to make it traversable, and to limit the game to 4 levels total.
- Files affected: `game.js`, `level_gen.js`
- Changes needed:
    - In `game.js`: Modify the `resetGame` function to loop back to Level 1 after Level 4 is completed, effectively removing levels 5+.
    - In `level_gen.js`: 
        - Update `generate` to apply the pre-calculated turn logic (previously only for Level 3) to Level 4 as well. This ensures walls are not generated on the side where the path turns.
        - Update `addWallWithGaps` to explicitly include Level 4 in the logic that adds gaps/holes to walls, fulfilling the user's request for more holes.
</analysis>

import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';
import { gameState, logGameInfo, logs } from './globals.js';
import { setupInput, updateInput } from './input.js';
import { setupRenderer, setupLighting } from './renderer.js';
import { setupUI, renderUI } from './ui.js';
import { LevelGenerator } from './level_gen.js';
import { PhysicsSystem } from './physics.js';
import { CameraController } from './camera.js';
import { PlayerBall } from './entities.js';

// Systems
const levelGenerator = new LevelGenerator();
const physicsSystem = new PhysicsSystem();
const cameraController = new CameraController();

function init() {
    setupRenderer();
    setupLighting();
    setupUI();
    setupInput();
    
    cameraController.setup();
    
    // Create initial dummy level or wait for start?
    // Let's create level 1
    startLevel(1);
    
    // Start Loop
    requestAnimationFrame(gameLoop);
    
    logGameInfo("INIT_COMPLETE");
}

function startLevel(index) {
    gameState.levelIndex = index;
    gameState.currentTilt.set(0, 0);
    gameState.shouldReset = false;
    
    // Generate Geometry
    levelGenerator.generate(index);
    
    // Create Player
    if (gameState.player) {
        // Cleanup old player mesh if exists (should be handled by level clear technically, but player is separate)
        gameState.worldContainer.remove(gameState.player.mesh);
    }
    gameState.player = new PlayerBall(gameState.playerSpawn);
    gameState.worldContainer.add(gameState.player.mesh);
    
    // Camera Reset
    cameraController.setup();
}

function resetGame() {
    if (gameState.gamePhase === "GAME_OVER_WIN") {
        let nextLevel = gameState.levelIndex + 1;
        // Cap at 4 levels, loop back to 1
        if (nextLevel > 4) {
            nextLevel = 1;
        }
        startLevel(nextLevel);
    } else {
        startLevel(gameState.levelIndex);
    }
    gameState.gamePhase = "PLAYING"; // Always auto-start after restart/next level
    gameState.shouldReset = false;
}

let lastTime = 0;

function gameLoop(timestamp) {
    requestAnimationFrame(gameLoop);
    
    const deltaTime = Math.min((timestamp - lastTime) / 1000, 0.1); // Cap dt
    lastTime = timestamp;
    
    gameState.deltaTime = deltaTime;
    gameState.frameCount++;
    
    // Logic
    if (gameState.shouldReset) {
        resetGame();
    }
    
    updateInput();
    
    if (gameState.gamePhase === "PLAYING") {
        physicsSystem.update(deltaTime);
        cameraController.update();
        
        if (gameState.goal) gameState.goal.update(timestamp / 1000);
        
        // Log player position periodically
        if (gameState.frameCount % 60 === 0 && gameState.player) {
            logs.player_info.push({
                x: gameState.player.position.x,
                y: gameState.player.position.y,
                z: gameState.player.position.z,
                tiltX: gameState.currentTilt.x,
                tiltZ: gameState.currentTilt.y,
                frame: gameState.frameCount
            });
        }
    } else if (gameState.gamePhase === "GAME_OVER_WIN" || gameState.gamePhase === "GAME_OVER_LOSE") {
         // Freeze is fine.
    }
    
    // Render
    gameState.renderer.render(gameState.scene, gameState.camera);
    renderUI();
}

// Start
init();

import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';
import { gameState, CONFIG } from './globals.js';
import { LevelBlock, GoalZone } from './entities.js';
import { RNG } from './utils.js';

export class LevelGenerator {
    constructor() {
        this.rng = null;
        this.levelIndex = 1;
    }

    generate(levelIndex) {
        this.levelIndex = levelIndex;
        // Change seed for Level 3 to avoid "spawn next to goal" looping issue
        const seed = levelIndex === 3 ? 1337 : levelIndex * 42;
        this.rng = new RNG(seed);
        
        // Cleanup existing
        this.clearLevel();
        
        // Init world container
        gameState.worldContainer = new THREE.Group();
        gameState.scene.add(gameState.worldContainer);
        
        // Level parameters
        // Make platforms bigger for levels > 2 without affecting 1 & 2
        const isEarlyLevel = levelIndex <= 2;
        const pathWidth = isEarlyLevel ? 12 : 18;
        const startSize = isEarlyLevel ? 12 : 20;
        const endSize = isEarlyLevel ? 14 : 24;

        const segmentCount = 5 + Math.floor(levelIndex * 1.5);
        let currentPos = new THREE.Vector3(0, 0, 0);
        let direction = new THREE.Vector3(0, 0, -1); // Forward
        
        // Start Platform
        this.createPlatform(currentPos, startSize, startSize, true);
        gameState.playerSpawn = currentPos.clone().add(new THREE.Vector3(0, 2, 0));
        
        // Generate path
        for (let i = 0; i < segmentCount; i++) {
            // Move forward
            const length = this.rng.rangeInt(15, 25);
            
            // For Level 3 and 4, we pre-calculate the turn to avoid blocking walls
            let turnDir = null;
            if (levelIndex >= 3 && i < segmentCount - 1) {
                const r = this.rng.random();
                if (r < 0.4) turnDir = 'LEFT';
                else if (r < 0.8) turnDir = 'RIGHT';
                else turnDir = 'NONE';
            }

            const nextPos = currentPos.clone().add(direction.clone().multiplyScalar(length));
            const midPos = currentPos.clone().lerp(nextPos, 0.5);
            
            // Create Corridor
            // Adjust midPos based on orientation to align boxes correctly
            if (Math.abs(direction.z) > 0.5) {
                // Moving Z
                this.createPlatform(midPos, pathWidth, length, false, turnDir, direction);
            } else {
                // Moving X
                this.createPlatform(midPos, length, pathWidth, false, turnDir, direction);
            }
            
            currentPos = nextPos;
            
            // Turn?
            if (i < segmentCount - 1) {
                if (levelIndex >= 3) {
                    // Use pre-calculated turn
                    if (turnDir === 'LEFT') {
                        const temp = direction.x; direction.x = direction.z; direction.z = -temp;
                    } else if (turnDir === 'RIGHT') {
                        const temp = direction.x; direction.x = -direction.z; direction.z = temp;
                    }
                } else {
                    // Original Logic for other levels (1 and 2)
                    const turn = this.rng.random();
                    if (turn < 0.4) {
                        // Turn Left
                        const temp = direction.x;
                        direction.x = direction.z;
                        direction.z = -temp;
                    } else if (turn < 0.8) {
                        // Turn Right
                        const temp = direction.x;
                        direction.x = -direction.z;
                        direction.z = temp;
                    }
                }
            }
        }
        
        // End Platform
        // Calculate entrance mask to skip walls blocking the path
        const skipMask = { negX: false, posX: false, negZ: false, posZ: false };
        if (direction.x > 0.5) skipMask.negX = true; // Moving East, enter from West
        else if (direction.x < -0.5) skipMask.posX = true; // Moving West, enter from East
        else if (direction.z > 0.5) skipMask.negZ = true; // Moving South, enter from North
        else if (direction.z < -0.5) skipMask.posZ = true; // Moving North, enter from South

        this.createPlatform(currentPos, endSize, endSize, true, null, null, skipMask);
        
        // Goal
        gameState.goal = new GoalZone(currentPos.clone().add(new THREE.Vector3(0, 0.6, 0)), 1.5);
        gameState.worldContainer.add(gameState.goal.mesh);
    }
    
    createPlatform(center, width, depth, isSafeZone, turnDir = null, moveDir = null, skipMask = null) {
        const thickness = 1;
        
        // Floor
        const floor = new LevelBlock(
            center, 
            new THREE.Vector3(width, thickness, depth), 
            'FLOOR'
        );
        this.addToLevel(floor);
        
        // Walls
        const wallHeight = 1.5;
        const wallThickness = 0.5;
        
        // Generalize wall placement logic:
        // If width > depth, it's X-aligned (walls on Z).
        // If depth >= width, it's Z-aligned (walls on X).
        const isXAligned = width > depth;

        if (isXAligned) {
            // X-Aligned Segment (Moving along X axis) -> Walls at +/- Z
            
            let skipPosZ = skipMask ? skipMask.posZ : false;
            let skipNegZ = skipMask ? skipMask.negZ : false;
            
            if (turnDir && moveDir) {
                if (moveDir.x > 0.5) { // Moving +X
                    // Left is -Z, Right is +Z
                    if (turnDir === 'LEFT') skipNegZ = true;
                    if (turnDir === 'RIGHT') skipPosZ = true;
                } else if (moveDir.x < -0.5) { // Moving -X
                    // Left is +Z, Right is -Z
                    if (turnDir === 'LEFT') skipPosZ = true;
                    if (turnDir === 'RIGHT') skipNegZ = true;
                }
            }

            const hasNegZ = !skipNegZ && (isSafeZone || this.rng.random() > 0.1);
            if (hasNegZ) {
                const pos = center.clone().add(new THREE.Vector3(0, wallHeight/2, -depth/2 - wallThickness/2));
                const size = new THREE.Vector3(width, wallHeight + thickness, wallThickness);
                this.addWallWithGaps(pos, size, 'X', isSafeZone);
            }

            const hasPosZ = !skipPosZ && (isSafeZone || this.rng.random() > 0.1);
            if (hasPosZ) {
                const pos = center.clone().add(new THREE.Vector3(0, wallHeight/2, depth/2 + wallThickness/2));
                const size = new THREE.Vector3(width, wallHeight + thickness, wallThickness);
                this.addWallWithGaps(pos, size, 'X', isSafeZone);
            }

        } else {
            // Z-Aligned Segment (Moving along Z axis) -> Walls at +/- X
            
            let skipPosX = skipMask ? skipMask.posX : false;
            let skipNegX = skipMask ? skipMask.negX : false;
            
            if (turnDir && moveDir) {
                // Moving Z
                if (moveDir.z < -0.5) { // North (-Z)
                    // Left is -X, Right is +X
                    if (turnDir === 'LEFT') skipNegX = true;
                    if (turnDir === 'RIGHT') skipPosX = true;
                } else if (moveDir.z > 0.5) { // South (+Z)
                    // Left is +X, Right is -X
                    if (turnDir === 'LEFT') skipPosX = true;
                    if (turnDir === 'RIGHT') skipNegX = true;
                }
            }

            const hasLeftWall = !skipNegX && (isSafeZone || this.rng.random() > 0.1);
            if (hasLeftWall) {
                const leftPos = center.clone().add(new THREE.Vector3(-width/2 - wallThickness/2, wallHeight/2, 0));
                const size = new THREE.Vector3(wallThickness, wallHeight + thickness, depth);
                this.addWallWithGaps(leftPos, size, 'Z', isSafeZone);
            }
            
            const hasRightWall = !skipPosX && (isSafeZone || this.rng.random() > 0.1);
            if (hasRightWall) {
                const rightPos = center.clone().add(new THREE.Vector3(width/2 + wallThickness/2, wallHeight/2, 0));
                const size = new THREE.Vector3(wallThickness, wallHeight + thickness, depth);
                this.addWallWithGaps(rightPos, size, 'Z', isSafeZone);
            }
        }
    }

    addWallWithGaps(pos, size, majorAxis, isSafeZone) {
        // For Level 3 and 4, introduce gaps in the walls to make it easier/possible
        if ((this.levelIndex === 3 || this.levelIndex === 4) && !isSafeZone && this.rng.random() > 0.3) {
            const gapSize = 4.0;
            
            if (majorAxis === 'X') {
                const len = size.x;
                if (len > gapSize * 2) {
                    const segmentLen = (len - gapSize) / 2;
                    const offset = (segmentLen + gapSize) / 2;
                    
                    // Left Segment
                    const p1 = pos.clone(); p1.x -= offset;
                    const s1 = size.clone(); s1.x = segmentLen;
                    this.addToLevel(new LevelBlock(p1, s1, 'WALL'));
                    
                    // Right Segment
                    const p2 = pos.clone(); p2.x += offset;
                    const s2 = size.clone(); s2.x = segmentLen;
                    this.addToLevel(new LevelBlock(p2, s2, 'WALL'));
                    return;
                }
            } else {
                // Z Axis
                const len = size.z;
                if (len > gapSize * 2) {
                    const segmentLen = (len - gapSize) / 2;
                    const offset = (segmentLen + gapSize) / 2;
                    
                    // Top Segment
                    const p1 = pos.clone(); p1.z -= offset;
                    const s1 = size.clone(); s1.z = segmentLen;
                    this.addToLevel(new LevelBlock(p1, s1, 'WALL'));
                    
                    // Bottom Segment
                    const p2 = pos.clone(); p2.z += offset;
                    const s2 = size.clone(); s2.z = segmentLen;
                    this.addToLevel(new LevelBlock(p2, s2, 'WALL'));
                    return;
                }
            }
        }
        
        // Default full wall
        this.addToLevel(new LevelBlock(pos, size, 'WALL'));
    }

    addToLevel(block) {
        gameState.levelObjects.push(block);
        gameState.worldContainer.add(block.mesh);
    }

    clearLevel() {
        if (gameState.worldContainer) {
            gameState.scene.remove(gameState.worldContainer);
        }
        gameState.levelObjects = [];
        gameState.goal = null;
    }
}
