<analysis>
- Issue identified: The current level generation logic creates a gap between every single platform, making the gameplay disjointed. The user requested fewer gaps, meaning platforms should often be contiguous or have very small gaps.
- Files affected: `world.js`
- Changes needed: 
  - Modified the loop in `setupLevel` to introduce a probability check (`isGap`) for creating significant gaps.
  - Set the default behavior to have a tiny gap (0.1) to connect platforms visually while keeping them as separate physics objects.
  - Reduced height variation for connected platforms to ensure they form climbable steps rather than insurmountable walls.
  - Retained larger gaps and height variations for the occasional "gap" segments to maintain some challenge.

import { Platform, Collectible } from './entities.js';
import { gameState, COLORS } from './globals.js';
import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';

export function setupLevel(levelIndex) {
    // Basic ground
    // Start platform
    new Platform(0, -1, 0, 10, 2, 5);
    
    // Procedurally generate a track
    let currentX = 5;
    let currentY = -1;
    
    const seededRandom = new Math.seedrandom(`level_${levelIndex}`);
    
    // Increase length slightly with level
    const length = 20 + Math.min(30, levelIndex * 2);
    
    for (let i = 0; i < length; i++) {
        // Difficulty scaling
        const difficulty = Math.min(1.0, levelIndex * 0.1);
        
        // Determine if this is a gap or a connected platform
        // 25% chance of a gap, increasing slightly with difficulty
        const isGap = seededRandom() < (0.25 + difficulty * 0.15);
        
        let gap = 0;
        let heightChange = 0;
        
        if (isGap) {
            // Standard gap logic
            gap = (1.5 + seededRandom() * 2.0) * (1 + difficulty * 0.2); 
            // Larger height changes allowed across gaps
            heightChange = (seededRandom() - 0.5) * (4 + difficulty * 2);
        } else {
            // Connected or tiny gap
            gap = 0.1; 
            // Smaller height changes for connected platforms to create slopes/steps
            heightChange = (seededRandom() - 0.5) * 2.0;
        }

        const width = Math.max(3.0, (4.5 + seededRandom() * 4.0) - difficulty);
        
        currentX += gap + width / 2;
        currentY += heightChange;
        
        // Clamp Y
        currentY = Math.max(-5, Math.min(5, currentY));
        
        new Platform(currentX, currentY, 0, width, 2, 4);
        
        // Add collectibles
        if (seededRandom() > 0.3) {
            new Collectible(currentX, currentY + 2.5, 0);
        }
        
        // Add hazards
        // Increase hazard chance with difficulty
        // Only place hazards on platforms if they are wide enough or if it's a gap destination
        if (i > 5 && seededRandom() > (0.7 - difficulty * 0.2)) {
            // Spikes or blocking block
             new Platform(currentX, currentY + 1.5, 0, 0.5, 1, 4, 'hazard');
        }
        
        currentX += width / 2;
    }
    
    // Goal Platform
    currentX += 5;
    new Platform(currentX, currentY, 0, 6, 2, 6, 'goal');
    
    // Background decoration (far away static meshes)
    for(let i=0; i<10 + levelIndex; i++) {
        const bgGeo = new THREE.BoxGeometry(2, 10 + Math.random() * 20, 2);
        const bgMat = new THREE.MeshStandardMaterial({ color: 0x111122 });
        const bg = new THREE.Mesh(bgGeo, bgMat);
        bg.position.set(i * 15 - 20, -10, -15);
        gameState.scene.add(bg);
        gameState.entities.push({ mesh: bg, update: () => {} }); // Passive entity
    }
}
